$ErrorActionPreference = 'Stop'

function Write-Diag {
  param([string]$Message, [ValidateSet('INFO','WARN','ERROR','DEBUG')][string]$Level='INFO')
  $line = "[{0}] [{1}] {2}" -f (Get-Date -Format "yyyy-MM-ddTHH:mm:ss"), $Level, $Message
  Write-Host $line
  try { Add-Content -LiteralPath $DiagLog -Value $line -ErrorAction Stop } catch {}
}

function TryGet([object]$obj, [string[]]$names) {
  if (-not $obj) { return $null }
  foreach ($n in $names) {
    if ($obj.PSObject.Properties.Name -contains $n) {
      $v = $obj.$n
      if ($null -ne $v -and "$v" -ne '') { return $v }
    }
  }
  return $null
}

function AsString([object]$v) {
  if ($null -eq $v) { return $null }
  $s = "$v".Trim()
  if ($s -eq '') { return $null }
  return $s
}

function AsUtc([object]$v) {
  try { if ($null -ne $v -and "$v" -ne '') { return ([datetime]$v).ToUniversalTime() } } catch {}
  return $null
}

function Get-GroupId($GroupObj) {
  if (-not $GroupObj) { return $null }
  return (AsString (TryGet $GroupObj @('Id','Identifier','MobilityGroupId','Uid')))
}

function Get-GroupName($GroupObj) {
  if (-not $GroupObj) { return $null }
  $n = AsString (TryGet $GroupObj @('Name','GroupName','DisplayName'))
  if ($n) { return $n }
  return (AsString (TryGet $GroupObj @('Id','Identifier')))
}

function Get-MigGroupId($Mig) {
  if (-not $Mig) { return $null }
  $gid = AsString (TryGet $Mig @('MobilityGroupId','MigrationGroupId'))
  if ($gid) { return $gid }
  $mg = TryGet $Mig @('MobilityGroup')
  if ($mg) { return AsString (TryGet $mg @('Id','Identifier')) }
  return $null
}

function Get-MigGroupName($Mig) {
  if (-not $Mig) { return $null }
  $gn = AsString (TryGet $Mig @('MobilityGroupName','GroupName'))
  if ($gn) { return $gn }
  $mg = TryGet $Mig @('MobilityGroup')
  if ($mg) { return AsString (TryGet $mg @('Name','GroupName','DisplayName')) }
  return $null
}

def _normalize_text(s: Any) -> str:
    return str(s).strip()

def _normalize_port(p: Any) -> Optional[str]:
    s = str(p).strip()
    if not s or s.lower() in {"nan", "none", "n/a"}:
        return None
    if s.endswith(".0"):
        s = s[:-2]
    return s

def find_col(df: pd.DataFrame, patterns: List[str]) -> Optional[str]:
    """Find a column whose name contains any of the patterns (case-insensitive)."""
    for col in df.columns:
        low = str(col).lower()
        for p in patterns:
            if p.lower() in low:
                return col
    return None

def _split_multi(s: str) -> List[str]:
    return [x.strip() for x in re.split(r"[,;\s]+", s or "") if x.strip()]

def _read_any_table(path: str) -> pd.DataFrame:
    p = Path(path)
    if not p.exists():
        raise FileNotFoundError(f"File not found: {p}")
    ext = p.suffix.lower()
    if ext == ".csv":
        return pd.read_csv(p)
    if ext in (".xlsx", ".xls"):
        return pd.read_excel(p)
    raise ValueError("Unsupported mapping file type. Use CSV or Excel.")

def _normalize_hostname_for_matching(name: Any) -> str:
    """Normalize hostname / asset tag for CMDB matching."""
    s = str(name).strip().upper()
    if not s:
        return ""
    if "." in s:
        s = s.split(".", 1)[0]
    return s

def _run_embedded_powershell(script_text: str, args: List[str], log_func=print) -> int:
    """
    Write script_text to a temp .ps1, execute PowerShell with args, delete temp file.
    Returns exit code.
    """
    if os.name != "nt":
        raise RuntimeError("PowerShell embedded scripts require Windows.")
    with tempfile.NamedTemporaryFile(delete=False, suffix=".ps1", mode="w", encoding="utf-8") as tmp:
        tmp.write(script_text)
        ps1_path = tmp.name

    cmd = ["powershell.exe", "-NoProfile", "-ExecutionPolicy", "Bypass", "-File", ps1_path] + args
    log_func(f"[PS] Running: {' '.join(cmd[:6])} ... (args hidden)")
    try:
        proc = subprocess.Popen(cmd)
        return proc.wait()
    finally:
        try:
            os.remove(ps1_path)
        except Exception:
            pass

def run_hcx_status_script(
    hcx_server: str,
    username: str,
    password: str,
    output_dir: str,
    group_ids: Optional[List[str]] = None,
    group_name_patterns: Optional[List[str]] = None,
    group_id_file: Optional[str] = None,
    log_func=print
) -> Dict[str, str]:
    if not hcx_server or not username or not password:
        raise ValueError("HCX server, username and password are required.")

    out_path = Path(output_dir).expanduser().resolve()
    out_path.mkdir(parents=True, exist_ok=True)

    args = ["-HcxServer", hcx_server, "-UserName", username, "-Password", password, "-OutputDir", str(out_path)]
    for gid in (group_ids or []):
        args += ["-IncludeGroupId", gid]
    for pat in (group_name_patterns or []):
        args += ["-IncludeGroupNameContains", pat]
    if group_id_file:
        args += ["-GroupIdFile", group_id_file]

    log_func(f"[HCX] Output directory: {out_path}")
    rc = _run_embedded_powershell(HCX_PS_SCRIPT, args, log_func=log_func)
    log_func(f"[HCX] PowerShell exit code: {rc}")
    if rc != 0:
        raise RuntimeError(f"HCX script failed with exit code {rc}. Check logs under {out_path}.")
    return {"returncode": str(rc), "output_dir": str(out_path)}

def run_ne_status_script(
    hcx_server: str,
    username: str,
    password: str,
    output_dir: str,
    ne_ids: Optional[List[str]] = None,
    name_patterns: Optional[List[str]] = None,
    log_func=print
) -> Dict[str, str]:
    if not hcx_server or not username or not password:
        raise ValueError("HCX server, username and password are required.")

    out_path = Path(output_dir).expanduser().resolve()
    out_path.mkdir(parents=True, exist_ok=True)

    run_tag = datetime.now().strftime("%Y%m%d_%H%M%S")
    args = [
        "-HcxServer", hcx_server,
        "-UserName", username,
        "-Password", password,
        "-OutputDir", str(out_path),
        "-RunTag", run_tag,
    ]
    for nid in (ne_ids or []):
        args += ["-IncludeId", nid]
    for pat in (name_patterns or []):
        args += ["-IncludeNameContains", pat]

    log_func(f"[NE] Output directory: {out_path}")
    log_func(f"[NE] Run tag: {run_tag}")
    rc = _run_embedded_powershell(NE_PS_SCRIPT, args, log_func=log_func)
    log_func(f"[NE] PowerShell exit code: {rc}")
    if rc != 0:
        raise RuntimeError(f"NE status script failed with exit code {rc}. Check logs under {out_path}.")

    run_dir = out_path / f"NE_Status_{run_tag}"
    return {
        "returncode": str(rc),
        "output_base_dir": str(out_path),
        "run_dir": str(run_dir),
        "raw_csv": str(run_dir / f"NE_Status_Raw_{run_tag}.csv"),
        "diag_log": str(run_dir / f"NE_Status_Diagnostic_{run_tag}.log"),
    }

def enrich_ne_status_with_vlan_mapping(
    ne_raw_csv: str,
    vlan_mapping_path: str,
    out_excel_path: str,
    log_func=print
) -> Dict[str, str]:
    """Merge NE status CSV with a CE-NE-VLAN mapping and write a single Excel."""
    status_df = pd.read_csv(ne_raw_csv)
    map_df = _read_any_table(vlan_mapping_path)

    # Heuristics for mapping columns
    name_col = None
    for c in map_df.columns:
        if str(c).strip().lower() == "name":
            name_col = c
            break
    if name_col is None:
        for c in map_df.columns:
            lc = str(c).lower()
            if "network" in lc or "vlan" in lc:
                name_col = c
                break

    ne_id_col = None
    for c in map_df.columns:
        if str(c).strip().lower() in {"ne_id", "ne", "network_extension", "network extension"}:
            ne_id_col = c
            break
    if ne_id_col is None:
        for c in map_df.columns:
            if "ne" in str(c).lower():
                ne_id_col = c
                break

    if name_col is None or ne_id_col is None:
        raise ValueError(f"Could not detect Name/NE columns in mapping. Columns: {list(map_df.columns)}")

    # Normalize join keys by network name
    status_df["__key"] = status_df.get("NetworkName", "").astype(str).str.strip().str.upper()
    map_df["__key"] = map_df[name_col].astype(str).str.strip().str.upper()

    # Optional placeholder drop
    if "IsPlaceholder" in map_df.columns:
        try:
            map_df = map_df[~map_df["IsPlaceholder"].fillna(False).astype(bool)].copy()
        except Exception:
            pass

    merged = map_df.merge(status_df, on="__key", how="left", suffixes=("_MAP", "_NE")).drop(columns=["__key"], errors="ignore")

    # Summary by NE + State
    if "State" in merged.columns:
        summary = (
            merged.groupby(ne_id_col)["State"]
            .value_counts(dropna=False)
            .unstack(fill_value=0)
            .reset_index()
        )
    else:
        summary = merged.groupby(ne_id_col).size().reset_index(name="RowCount")

    log_func(f"[NE] Writing enriched Excel: {out_excel_path}")
    with pd.ExcelWriter(out_excel_path, engine="openpyxl") as writer:
        status_df.drop(columns=["__key"], errors="ignore").to_excel(writer, sheet_name="NE_Status_Raw", index=False)
        map_df.drop(columns=["__key"], errors="ignore").to_excel(writer, sheet_name="VLAN_Mapping", index=False)
        merged.to_excel(writer, sheet_name="NE_VLAN_Status", index=False)
        summary.to_excel(writer, sheet_name="Summary", index=False)

    return {"enriched_excel": out_excel_path}

def _read_dependency_export(input_path: str) -> pd.DataFrame:
    """
    Read an Azure Migrate dependency export (CSV/XLSX) and return a DataFrame
    with the canonical columns expected by the rest of the dependency pipeline:

        - 'Source server name'
        - 'Destination server name'
        - 'Destination port'

    IMPORTANT:
      - We keep the original behavior if the file already has the exact required columns.
      - If not, we attempt a safe, backwards-compatible header mapping (case/format variations),
        e.g. Azure Migrate exports like:
           'Source Server Name', 'Destination Server Name', 'Destination Port'
    """
    _, ext = os.path.splitext(input_path.lower())
    if ext == ".csv":
        df = pd.read_csv(input_path)
    elif ext in (".xlsx", ".xls"):
        df = pd.read_excel(input_path)
    else:
        raise ValueError("Unsupported file type. Select a CSV or Excel file.")

    # Normalize header whitespace (Azure Migrate export headers sometimes vary by casing/spacing)
    df.columns = [str(c).strip() for c in df.columns]

    # ---- Fast path: exact legacy headers present (do not change what already works) ----
    if all(c in df.columns for c in REQUIRED_COLUMNS):
        src_col, dst_col, port_col = REQUIRED_COLUMNS

    # ---- Flexible path: map newer/different Azure Migrate headers -> canonical headers ----
    else:
        def _find_cols(patterns):
            pats = [p.lower() for p in patterns]
            matches = []
            for col in df.columns:
                low = str(col).lower()
                if any(p in low for p in pats):
                    matches.append(col)
            return matches

        def _pick_best(pattern_groups, banned_terms=None):
            """Pick the best matching column given priority pattern groups."""
            banned_terms = [b.lower() for b in (banned_terms or [])]

            for group in pattern_groups:
                if isinstance(group, str):
                    group = [group]

                matches = _find_cols(group)
                if not matches:
                    continue

                # If there are multiple candidates, try to drop clearly-wrong ones (e.g., ARM IDs)
                if banned_terms and len(matches) > 1:
                    filtered = [c for c in matches if not any(b in str(c).lower() for b in banned_terms)]
                    if filtered:
                        matches = filtered

                # Prefer an exact (case-insensitive) header match first
                for p in group:
                    exact = [c for c in matches if str(c).lower().strip() == p.lower().strip()]
                    if exact:
                        return sorted(exact, key=lambda c: len(str(c)))[0]

                # Otherwise, choose the shortest match (usually the cleanest header)
                return sorted(matches, key=lambda c: len(str(c)))[0]

            return None

        # Prefer '... Server Name' over '... ARM ID' when both exist
        src_col = _pick_best(
            [
                "source server name",
                "source machine name",
                "source host name",
                "source hostname",
                "source host",
                "source machine",
            ],
            banned_terms=["arm id"],
        )
        dst_col = _pick_best(
            [
                "destination server name",
                "destination machine name",
                "destination host name",
                "destination hostname",
                "destination host",
                "destination machine",
            ],
            banned_terms=["arm id"],
        )
        port_col = _pick_best(
            [
                "destination port",
                "dest port",
                "remote port",
                "port",
            ]
        )

        if not (src_col and dst_col and port_col):
            raise ValueError(
                f"Missing required columns: {REQUIRED_COLUMNS}. Found: {list(df.columns)}"
            )

    # Build canonical DF for the rest of the pipeline
    df = df.dropna(subset=[src_col, dst_col]).copy()
    df = df.rename(
        columns={
            src_col: "Source server name",
            dst_col: "Destination server name",
            port_col: "Destination port",
        }
    )

    df["Source server name"] = df["Source server name"].map(_normalize_text)
    df["Destination server name"] = df["Destination server name"].map(_normalize_text)
    df["Destination port"] = df["Destination port"].map(_normalize_port)

    df = df.dropna(subset=["Destination port"]).copy()
    df["Destination port"] = df["Destination port"].astype(str)
    return df

def _read_dependency_exports(input_paths: Any) -> pd.DataFrame:
    if isinstance(input_paths, (str, Path)):
        paths = [str(input_paths)]
    else:
        paths = [str(p) for p in input_paths]
    if not paths:
        raise ValueError("No dependency files provided.")
    dfs = [_read_dependency_export(p) for p in paths]
    return pd.concat(dfs, ignore_index=True)

def process_csv_and_generate_output_df(df: pd.DataFrame, output_csv_basepath: str):
    df = df.copy()
    df['Connection Count'] = df.groupby('Source server name')['Destination server name'].transform('nunique')
    df = df.sort_values(by=['Connection Count', 'Source server name'])

    unique_rows = set()
    rows = []
    for _, r in df.iterrows():
        key = (r['Source server name'], r['Destination server name'], r['Destination port'])
        if key not in unique_rows:
            unique_rows.add(key)
            rows.append({
                'Source server name': r['Source server name'],
                'Destination server name': r['Destination server name'],
                'Destination port': r['Destination port'],
                'Connection Count': r['Connection Count']
            })

    expanded_df = pd.DataFrame(rows, columns=['Source server name', 'Destination server name', 'Destination port', 'Connection Count'])
    connection_counts = expanded_df['Connection Count'].value_counts().sort_index()

    output_csv_path = f"{output_csv_basepath}_grouped.csv"
    output_viz_path = f"{output_csv_basepath}_connections_viz.png"

    plt.figure(figsize=(10, 6))
    connection_counts.plot(kind='bar')
    plt.title('Number of Source Servers per Connection Count')
    plt.xlabel('Number of Connections')
    plt.ylabel('Number of Source Servers')
    plt.tight_layout()
    plt.savefig(output_viz_path)
    plt.close()

    expanded_df.to_csv(output_csv_path, index=False)
    return expanded_df, output_csv_path, output_viz_path

def write_all_unique_ports(expanded_df: pd.DataFrame, ports_csv_path: str) -> List[str]:
    ports = sorted({str(p).strip() for p in expanded_df['Destination port'].dropna().astype(str)})
    pd.DataFrame([",".join(ports)], columns=["Destination port"]).to_csv(ports_csv_path, index=False)
    return ports

def determine_complexity(num_servers: int) -> str:
    if num_servers <= 2:
        return 'Low'
    elif 3 <= num_servers <= 4:
        return 'Medium'
    else:
        return 'Complex'

def group_by_source_all_ports_with_groups(input_grouped_csv_path: str):
    df = pd.read_csv(input_grouped_csv_path, dtype={"Destination port": str})
    df['Source server name'] = df['Source server name'].map(_normalize_text)
    df['Destination server name'] = df['Destination server name'].map(_normalize_text)
    df['Destination port'] = df['Destination port'].astype(str).str.strip()

    edge_ports = defaultdict(set)
    source_to_dests = defaultdict(set)

    for _, r in df.iterrows():
        src = r['Source server name']
        dst = r['Destination server name']
        port = r['Destination port']
        source_to_dests[src].add(dst)
        edge_ports[(src, dst)].add(port)

    rows = []
    for src, dests in source_to_dests.items():
        dest_list = sorted(dests)
        total_servers = 1 + len(dest_list)
        rows.append({
            'Application Server': src,
            'Dependency Servers': dest_list,
            'Total Servers': total_servers,
            'Complexity': determine_complexity(total_servers)
        })

    grouped_df = pd.DataFrame(rows, columns=['Application Server', 'Dependency Servers', 'Total Servers', 'Complexity'])
    order = {'Low': 1, 'Medium': 2, 'Complex': 3}
    grouped_df['Complexity Order'] = grouped_df['Complexity'].map(order)
    grouped_df = grouped_df.sort_values(by=['Complexity Order', 'Total Servers', 'Application Server'], ascending=[True, True, True]).drop(columns='Complexity Order').reset_index(drop=True)

    sig_to_group = OrderedDict()
    groups = []
    for _, row in grouped_df.iterrows():
        signature = tuple(row['Dependency Servers'])
        if signature not in sig_to_group:
            sig_to_group[signature] = f"Group {len(sig_to_group) + 1}"
        groups.append(sig_to_group[signature])
    grouped_df.insert(0, 'Group', groups)

    base = os.path.splitext(input_grouped_csv_path)[0]
    complexity_csv_path = base + '_ComplexityGrouping.csv'
    grouped_df.to_csv(complexity_csv_path, index=False)

    app_to_group = {row['Application Server']: row['Group'] for _, row in grouped_df.iterrows()}
    return grouped_df, complexity_csv_path, edge_ports, app_to_group

def inject_search_control(html_path: str):
    """Inject a small search box into the vis-network HTML (search labels/titles)."""
    try:
        with open(html_path, "r", encoding="utf-8") as f:
            html = f.read()

        search_html = r"""
<div style="position:fixed; top:10px; right:10px; z-index:9999; background:#ffffff; border:1px solid #ccc; padding:6px; font-family:Segoe UI, Arial, sans-serif; font-size:12px;">
  <label for="nodeSearchInput" style="margin-right:4px;">Find node / group:</label>

</div>
<script type="text/javascript">
function searchNode() {
  var termBox = document.getElementById('nodeSearchInput');
  if (!termBox) return;
  var term = termBox.value.toLowerCase();
  if (!term) return;

  var foundIds = [];
  nodes.get().forEach(function(n) {
    var label = String(n.label || "").toLowerCase();
    var title = String(n.title || "").toLowerCase();
    if (label.indexOf(term) !== -1 || title.indexOf(term) !== -1) {

        with open(html_path, "w", encoding="utf-8") as f:
            f.write(html)
    except Exception as e:

def create_interactive_network_graph(grouped_df: pd.DataFrame, edge_ports_map: Dict[Tuple[str, str], set], app_to_group: Dict[str, str], output_html_path: str):
    """
    Build an interactive HTML graph using vis-network directly (no pyvis dependency).
    """
    import json

    nodes_dict: Dict[str, Dict[str, str]] = {}
    edges_dict: Dict[Tuple[str, str], Dict[str, str]] = {}

    for _, row in grouped_df.iterrows():
        app = row["Application Server"]
        dep_list = row["Dependency Servers"]
        group_label = row["Group"]

        app_node_id = f"{group_label} | {app}"
        if app_node_id not in nodes_dict:
            nodes_dict[app_node_id] = {"id": app_node_id, "label": app, "title": app, "group": group_label}

        for dep in dep_list:
            dep_node_id = f"{group_label} | {dep}"
            if dep_node_id not in nodes_dict:
                nodes_dict[dep_node_id] = {"id": dep_node_id, "label": dep, "title": dep, "group": group_label}

            ports = sorted(edge_ports_map.get((app, dep), set()), key=lambda x: (0, int(x)) if str(x).isdigit() else (1, str(x)))
            port_text = ", ".join(ports) if ports else "N/A"

            edge_key = tuple(sorted([app_node_id, dep_node_id]))
            if edge_key not in edges_dict:
                edges_dict[edge_key] = {"from": app_node_id, "to": dep_node_id, "title": f"Ports: {port_text}", "label": port_text}

    nodes = list(nodes_dict.values())
    edges = list(edges_dict.values())

    options = {
        "interaction": {"hover": True},
        "physics": {"enabled": True, "stabilization": {"enabled": True, "iterations": 200}},
        "configure": {"enabled": True},
        "nodes": {"shape": "dot", "size": 16},
        "edges": {"smooth": True},
    }

    html_template = r"""
<!DOCTYPE html>
<html>
<head>
  <meta charset="utf-8" />
  <title>CAFToolKit Dependency Map</title>
  <style type="text/css">
    #mynetwork {{
      width: 1500px;
      height: 1500px;
      border: 1px solid lightgray;
    }}
  </style>
  <script type="text/javascript"
          src="https://unpkg.com/vis-network/standalone/umd/vis-network.min.js">
  </script>
</head>
<body>
<div id="mynetwork"></div>

<script type="text/javascript">
  var nodes = new vis.DataSet({nodes_json});
  var edges = new vis.DataSet({edges_json});

  var container = document.getElementById('mynetwork');
  var data = {{nodes: nodes, edges: edges}};
  var options = {options_json};

  var network = new vis.Network(container, data, options);
</script>
</body>
</html>
"""
    html = html_template.format(nodes_json=json.dumps(nodes), edges_json=json.dumps(edges), options_json=json.dumps(options))
    output_html_path = os.path.abspath(output_html_path)
    with open(output_html_path, "w", encoding="utf-8") as f:
        f.write(html)
    inject_search_control(output_html_path)

def _load_cmdb(cmdb_path: str) -> pd.DataFrame:
    _, ext = os.path.splitext(cmdb_path.lower())
    if ext == ".csv":
        try:
            df = pd.read_csv(cmdb_path)
        except UnicodeDecodeError:
            df = pd.read_csv(cmdb_path, encoding="ISO-8859-1")
    elif ext in (".xlsx", ".xls"):
        df = pd.read_excel(cmdb_path)
    else:
        raise ValueError("Unsupported CMDB file type. Please use CSV or Excel.")

    if df.empty:
        raise ValueError("CMDB file appears to be empty.")

    cols = list(df.columns)
    col_map = {str(c).strip().lower(): c for c in cols}

    asset_col = None
    for key in col_map:
        if "asset tag" in key or key == "server" or key == "hostname" or "ci name" in key:
            asset_col = col_map[key]
            break
    if asset_col is None and "Asset Tag" in df.columns:
        asset_col = "Asset Tag"
    if asset_col is None:
        raise ValueError("CMDB file must have a column like 'Asset Tag' / 'Server' / 'Hostname'.")

    app_col = None
    for key in col_map:
        if "bsp name" in key or "application" in key or "app name" in key:
            app_col = col_map[key]
            break
    if app_col is None and "BSP Name" in df.columns:
        app_col = "BSP Name"
    if app_col is None:
        raise ValueError("CMDB file must have an application column such as 'BSP Name' / 'Application'.")

    df = df[[asset_col, app_col]].copy()
    df.rename(columns={asset_col: "AssetTag", app_col: "Application"}, inplace=True)
    df["AssetTag_norm"] = df["AssetTag"].astype(str).str.strip().str.upper()
    df["Application"] = df["Application"].astype(str).str.strip()
    df = df.dropna(subset=["AssetTag_norm", "Application"])
    return df

def group_applications_from_edges(edge_ports_map: Dict[Tuple[str, str], set], output_base: str):
    source_to_dests = defaultdict(set)
    for (src_app, dst_app), _ports in edge_ports_map.items():
        source_to_dests[src_app].add(dst_app)

    rows = []
    for src_app, dests in source_to_dests.items():
        dest_list = sorted(dests)
        total = 1 + len(dest_list)
        rows.append({"Application": src_app, "Dependency Applications": dest_list, "Total Applications": total, "Complexity": determine_complexity(total)})

    grouped_df = pd.DataFrame(rows, columns=["Application", "Dependency Applications", "Total Applications", "Complexity"])
    if grouped_df.empty:
        complexity_csv_path = f"{output_base}_AppComplexityGrouping.csv"
        grouped_df.to_csv(complexity_csv_path, index=False)
        return grouped_df, complexity_csv_path, {}

    order = {"Low": 1, "Medium": 2, "Complex": 3}
    grouped_df["Complexity Order"] = grouped_df["Complexity"].map(order)
    grouped_df = grouped_df.sort_values(by=["Complexity Order", "Total Applications", "Application"], ascending=[True, True, True]).drop(columns="Complexity Order").reset_index(drop=True)

    sig_to_group = OrderedDict()
    groups = []
    for _, row in grouped_df.iterrows():
        signature = tuple(row["Dependency Applications"])
        if signature not in sig_to_group:
            sig_to_group[signature] = f"Group {len(sig_to_group) + 1}"
        groups.append(sig_to_group[signature])
    grouped_df.insert(0, "Group", groups)

    complexity_csv_path = f"{output_base}_AppComplexityGrouping.csv"
    grouped_df.to_csv(complexity_csv_path, index=False)
    app_to_group = {row["Application"]: row["Group"] for _, row in grouped_df.iterrows()}
    return grouped_df, complexity_csv_path, app_to_group

def run_full_dep_pipeline(input_paths: Any, log_func=print) -> Dict[str, str]:
    cwd = os.getcwd()
    paths = [str(input_paths)] if isinstance(input_paths, (str, Path)) else [str(p) for p in input_paths]
    if not paths:
        raise ValueError("No dependency files provided.")

    base_stem = os.path.splitext(os.path.basename(paths[0]))[0] + ("_merged" if len(paths) > 1 else "")
    output_base = os.path.join(cwd, base_stem)

    log_func(f"[DEP] [1/6] Reading {len(paths)} Azure Migrate dependency file(s)…")
    for p in paths:
        log_func(f"[DEP]      - {p}")
    raw_df = _read_dependency_exports(paths)
    log_func(f"[DEP]      - Combined rows: {len(raw_df)}")

    log_func(f"[DEP] [2/6] De-dup + grouped connections & bar chart…")
    expanded_df, grouped_csv_path, viz_path = process_csv_and_generate_output_df(raw_df, output_base)
    log_func(f"[DEP]      - Grouped CSV: {grouped_csv_path}")
    log_func(f"[DEP]      - Chart PNG : {viz_path}")

    log_func(f"[DEP] [3/6] Writing ALL unique ports (reference)…")
    ports_all_csv_path = os.path.splitext(grouped_csv_path)[0] + '_unique_ports.csv'
    all_ports = write_all_unique_ports(expanded_df, ports_all_csv_path)
    log_func(f"[DEP]      - {len(all_ports)} unique ports")
    log_func(f"[DEP]      - Ports CSV : {ports_all_csv_path}")

    log_func(f"[DEP] [4/6] Grouping by Source across ALL ports + Group numbering…")
    grouped_df, complexity_csv_path, edge_ports, app_to_group = group_by_source_all_ports_with_groups(grouped_csv_path)
    log_func(f"[DEP]      - Complexity CSV: {complexity_csv_path}")
    log_func(f"[DEP]      - Groups discovered: {len(set(grouped_df['Group']))}")

    log_func(f"[DEP] [5/6] Building interactive network (HTML)…")
    html_path = os.path.splitext(grouped_csv_path)[0] + '_network.html'
    create_interactive_network_graph(grouped_df, edge_ports, app_to_group, html_path)
    log_func(f"[DEP]      - HTML graph : {html_path}")

    log_func(f"[DEP] [6/6] Opening interactive graph in browser…")
    webbrowser.open('file://' + os.path.abspath(html_path))

    log_func("[DEP] === Dependency Mapping DONE (HOST-BASED) ===")
    return {
        "mode": "HOST_ONLY",
        "input_files": " ; ".join(paths),
        "grouped_csv": grouped_csv_path,
        "ports_all_csv": ports_all_csv_path,
        "complexity_csv": complexity_csv_path,
        "chart_png": viz_path,
        "network_html": html_path
    }

def run_full_dep_pipeline_with_cmdb(input_paths: Any, cmdb_path: str, log_func=print) -> Dict[str, str]:
    cwd = os.getcwd()
    paths = [str(input_paths)] if isinstance(input_paths, (str, Path)) else [str(p) for p in input_paths]
    if not paths:
        raise ValueError("No dependency files provided.")

    base_stem = os.path.splitext(os.path.basename(paths[0]))[0] + ("_merged_CMDB" if len(paths) > 1 else "_CMDB")
    output_base = os.path.join(cwd, base_stem)

    log_func(f"[DEP+CMDB] [1/6] Reading {len(paths)} Azure Migrate dependency file(s)…")
    raw_df = _read_dependency_exports(paths)
    log_func(f"[DEP+CMDB]      - Combined rows: {len(raw_df)}")

    log_func(f"[DEP+CMDB] [2/6] Loading CMDB file: {cmdb_path}")
    cmdb_df = _load_cmdb(cmdb_path)

    server_to_app: Dict[str, str] = {}
    for _, row in cmdb_df.iterrows():
        key = _normalize_hostname_for_matching(row["AssetTag"])
        if key:
            server_to_app[key] = row["Application"]
    log_func(f"[DEP+CMDB]      - CMDB entries with application: {len(server_to_app)}")

    raw_df = raw_df.copy()
    raw_df["Source_norm"] = raw_df["Source server name"].map(_normalize_hostname_for_matching)
    raw_df["Dest_norm"] = raw_df["Destination server name"].map(_normalize_hostname_for_matching)

    app_edge_ports = defaultdict(set)
    total_rows = len(raw_df)
    mapped_rows = 0
    for _, r in raw_df.iterrows():
        src_app = server_to_app.get(r["Source_norm"])
        dst_app = server_to_app.get(r["Dest_norm"])
        if src_app and dst_app:
            mapped_rows += 1
            port = str(r["Destination port"]).strip()
            app_edge_ports[(src_app, dst_app)].add(port)

    log_func(f"[DEP+CMDB] [3/6] Mapped {mapped_rows} of {total_rows} rows to application->application edges.")
    if not app_edge_ports:
        raise ValueError("No application-to-application mappings generated. Check CMDB vs server names.")

    edges_rows = [{"Source Application": k[0], "Destination Application": k[1], "Ports": ",".join(sorted(v))} for k, v in app_edge_ports.items()]
    edges_df = pd.DataFrame(edges_rows, columns=["Source Application", "Destination Application", "Ports"])
    edges_csv_path = f"{output_base}_AppEdges.csv"
    edges_df.to_csv(edges_csv_path, index=False)
    log_func(f"[DEP+CMDB]      - App edges CSV: {edges_csv_path}")

    log_func(f"[DEP+CMDB] [4/6] Grouping applications into waves based on connectivity…")
    grouped_df_apps, complexity_csv_path, app_to_group = group_applications_from_edges(app_edge_ports, output_base)
    log_func(f"[DEP+CMDB]      - App complexity CSV: {complexity_csv_path}")

    html_path = ""
    if not grouped_df_apps.empty:
        host_style_df = grouped_df_apps.rename(columns={"Application": "Application Server", "Dependency Applications": "Dependency Servers"})

        html_path = f"{output_base}_AppNetwork.html"
        create_interactive_network_graph(host_style_df, app_edge_ports, app_to_group, html_path)
        log_func(f"[DEP+CMDB]      - HTML app graph: {html_path}")

    return {

    }

def load_rvtools(path: Path) -> Dict[str, pd.DataFrame]:
    xl = pd.ExcelFile(path)
    return {sheet: xl.parse(sheet) for sheet in xl.sheet_names}

def bool_col(df: pd.DataFrame, col_name: str, default: bool = False) -> pd.Series:
    if col_name in df.columns:
        return df[col_name].fillna(False).astype(bool)
    return pd.Series([default] * len(df), index=df.index)

def check_powerstate(vinfo: pd.DataFrame) -> Dict[str, Dict[str, bool]]:
    col = find_col(vinfo, ["powerstate"])
    guest_col = find_col(vinfo, ["guest state"])
    hb_col = find_col(vinfo, ["heartbeat"])
    cfg_col = find_col(vinfo, ["config status"])

    results = {}
    for _, row in vinfo.iterrows():
        vm = str(row["VM"]).strip()
        powered_on = bool(col and str(row[col]).lower() == "poweredon")
        guest_running = bool(guest_col and str(row[guest_col]).lower() == "running")
        hb_ok = bool(hb_col and str(row[hb_col]).lower() in ["green"])
        cfg_ok = bool(cfg_col and str(row[cfg_col]).lower() == "green")
        results[vm] = {"powered_on": powered_on, "guest_running": guest_running, "heartbeat_ok": hb_ok, "config_ok": cfg_ok}
    return results

def check_vmtools(vtools: Optional[pd.DataFrame]) -> Dict[str, Dict[str, Any]]:
    if vtools is None or vtools.empty:
        return {}
    tools_col = find_col(vtools, ["tools"])
    ver_col = find_col(vtools, ["tools version"])
    req_col = find_col(vtools, ["required"])
    results = {}

    for _, row in vtools.iterrows():
        vm = str(row["VM"]).strip()
        tools_state = str(row[tools_col]).lower() if tools_col else ""
        healthy = (tools_state == "toolsok")
        outdated = False
        if ver_col and req_col:
            try:
                outdated = int(row[ver_col]) < int(row[req_col])
            except Exception:
                outdated = False
        results[vm] = {"tools_healthy": healthy and not outdated, "tools_state": tools_state, "tools_outdated": outdated}
    return results

def check_mounted_iso(vcd: Optional[pd.DataFrame], vusb: Optional[pd.DataFrame]) -> Dict[str, bool]:
    results = {}
    if vcd is not None and not vcd.empty:
        for _, row in vcd.iterrows():
            vm = str(row.get("VM", "")).strip()
            connected = bool(row.get("Connected", False)) or bool(row.get("Starts Connected", False))
            if vm and connected:
                results[vm] = True
    if vusb is not None and not vusb.empty:
        for _, row in vusb.iterrows():
            vm = str(row.get("VM", "")).strip()
            if vm and bool(row.get("Connected", False)):
                results[vm] = True
    return results

def check_disk_multiple_1024(vdisk: Optional[pd.DataFrame]) -> Dict[str, Any]:
    if vdisk is None or vdisk.empty:
        return {}
    cap_col = find_col(vdisk, ["capacity mib"])
    if not cap_col:
        return {"_error": "No Capacity column found in vDisk."}
    results: Dict[str, List[Dict[str, Any]]] = {}
    for _, row in vdisk.iterrows():
        vm = str(row.get("VM", "")).strip()
        disk = row.get("Disk")
        try:
            size = int(row[cap_col])
            aligned = (size % 1024 == 0)
        except Exception:
            aligned = False
        if vm and not aligned:
            results.setdefault(vm, []).append({"disk": disk, "size_mib": row[cap_col]})
    return results

def check_partition_free(vpart: Optional[pd.DataFrame], threshold: float = 20.0) -> Dict[str, Any]:
    if vpart is None or vpart.empty:
        return {}
    disk_col = find_col(vpart, ["disk"])
    free_col = find_col(vpart, ["free %"])
    if not disk_col or not free_col:
        return {}
    results: Dict[str, List[Dict[str, Any]]] = {}
    for _, row in vpart.iterrows():
        vm = str(row.get("VM", "")).strip()
        disk = row.get(disk_col)
        try:
            freepct = float(row[free_col])
        except Exception:
            continue
        if vm and freepct < threshold:
            results.setdefault(vm, []).append({"partition": disk, "free_pct": freepct})
    return results

def check_recent_backup(vinfo: pd.DataFrame, max_age_days: int = 7) -> Dict[str, Any]:
    col = find_col(vinfo, ["backup"])
    if not col:
        return {"_error": "Backup info not present in RVTools"}
    results = {}
    now = datetime.now(timezone.utc)
    for _, row in vinfo.iterrows():
        vm = str(row.get("VM", "")).strip()
        val = str(row.get(col, ""))
        if not vm:
            continue
        if val.lower() == "nan" or val.strip() == "":
            results[vm] = {"has_backup": False, "message": "No backup info"}
            continue
        match = re.search(r'at (.*)', val)
        if not match:
            results[vm] = {"has_backup": False, "message": "Unparsable backup info"}
            continue
        try:
            dt = pd.to_datetime(match.group(1), utc=True)
        except Exception:
            results[vm] = {"has_backup": False, "message": "Failed to parse backup date"}
            continue
        age_days = (now - dt).days
        ok = age_days <= max_age_days
        results[vm] = {"has_backup": ok, "age_days": age_days, "backup_datetime": dt}
    return results

def check_snapshots(vsnap: Optional[pd.DataFrame]) -> Dict[str, Any]:
    if vsnap is None or vsnap.empty:
        return {}
    results: Dict[str, List[Dict[str, Any]]] = {}
    for _, row in vsnap.iterrows():
        vm = str(row.get("VM", "")).strip()
        if not vm:
            continue
        results.setdefault(vm, []).append({
            "snapshot_name": row.get("Name"),
            "date": row.get("Date / time"),
            "size_total_mib": row.get("Size MiB (total)")
        })
    return results

def extract_vm_networks_from_rvtools(dfs: Dict[str, pd.DataFrame]) -> Dict[str, List[str]]:
    vnet = dfs.get("vNetwork")
    if vnet is None or vnet.empty:
        return {}
    vnet = vnet.copy()
    vm_col = find_col(vnet, ["vm", "name"])
    if not vm_col:
        return {}
    net_col = None
    for patt in ["portgroup", "port group", "network", "dvportgroup", "dv portgroup"]:
        net_col = find_col(vnet, [patt])
        if net_col:
            break
    if not net_col:
        return {}
    vnet["VM"] = vnet[vm_col].astype(str).str.strip()
    vnet["NetworkName"] = vnet[net_col].astype(str).str.strip()
    vnet = vnet[(vnet["VM"] != "") & (vnet["NetworkName"] != "")]
    vm_to_nets = {}
    for vm, sub in vnet.groupby("VM"):
        nets = sorted({str(x).strip() for x in sub["NetworkName"] if str(x).strip()})
        vm_to_nets[vm] = nets
    return vm_to_nets

def run_rvtools_checks(dfs: Dict[str, pd.DataFrame]):
    vinfo = dfs.get("vInfo")
    if vinfo is None or vinfo.empty:
        raise ValueError("RVTools workbook does not contain a non-empty vInfo sheet.")

    vtools = dfs.get("vTools")
    vcd = dfs.get("vCD")
    vusb = dfs.get("vUSB")
    vdisk = dfs.get("vDisk")
    vpart = dfs.get("vPartition")
    vsnap = dfs.get("vSnapshot")

    vinfo = vinfo.copy()
    vinfo["VM"] = vinfo["VM"].astype(str).str.strip()
    vms = sorted(vinfo["VM"].unique())
    summary = pd.DataFrame(index=vms)

    power = check_powerstate(vinfo)
    tools = check_vmtools(vtools)
    iso = check_mounted_iso(vcd, vusb)
    disks = check_disk_multiple_1024(vdisk)
    free = check_partition_free(vpart)
    backup = check_recent_backup(vinfo)
    snaps = check_snapshots(vsnap)

    summary["Powered On"] = [power.get(vm, {}).get("powered_on", False) for vm in vms]
    summary["Guest Running"] = [power.get(vm, {}).get("guest_running", False) for vm in vms]
    summary["Heartbeat OK"] = [power.get(vm, {}).get("heartbeat_ok", False) for vm in vms]
    summary["Config OK"] = [power.get(vm, {}).get("config_ok", False) for vm in vms]
    summary["VMTools Healthy"] = [tools.get(vm, {}).get("tools_healthy", False) for vm in vms]
    summary["ISO/Device Attached"] = [vm in iso for vm in vms]
    summary["Disk Misaligned"] = [vm in disks for vm in vms if vm != "_error"]
    summary["Low Free Partitions (<20%)"] = [vm in free for vm in vms]
    summary["Backup OK"] = [backup.get(vm, {}).get("has_backup", False) for vm in vms]
    summary["Has Snapshots"] = [vm in snaps for vm in vms]

    return summary, disks, free, iso, snaps

def add_ready_and_reasons(summary: pd.DataFrame) -> pd.DataFrame:
    powered_on = bool_col(summary, "Powered On")
    guest_running = bool_col(summary, "Guest Running")
    heartbeat_ok = bool_col(summary, "Heartbeat OK")
    config_ok = bool_col(summary, "Config OK")
    tools_ok = bool_col(summary, "VMTools Healthy")
    no_iso = ~bool_col(summary, "ISO/Device Attached")
    no_misaligned = ~bool_col(summary, "Disk Misaligned")
    no_lowspace = ~bool_col(summary, "Low Free Partitions (<20%)")
    no_snapshots = ~bool_col(summary, "Has Snapshots")

    # Backup is warning-only (not gating)
    summary["ReadyToMigrate"] = (powered_on & guest_running & heartbeat_ok & config_ok & tools_ok & no_iso & no_misaligned & no_lowspace & no_snapshots)

    reasons = []
    for idx in summary.index:
        r = []
        if not powered_on[idx]:
            r.append("VM is not powered on")
        if not guest_running[idx]:
            r.append("Guest OS not running")
        if not heartbeat_ok[idx]:
            r.append("Heartbeat not green")
        if not config_ok[idx]:
            r.append("VM config not healthy")
        if not tools_ok[idx]:
            r.append("VMTools not healthy / outdated")
        if not no_iso[idx]:
            r.append("Mounted ISO or attached device")
        if not no_misaligned[idx]:
            r.append("Disk not multiple of 1024")
        if not no_lowspace[idx]:
            r.append("Low free disk space (<20%)")
        if not no_snapshots[idx]:
            r.append("Snapshots exist")
        reasons.append("; ".join(r) if r else "")
    summary["ReasonNotReady"] = reasons
    return summary

def load_scoped(scope_path: Path, sheet_name: Optional[str] = None) -> pd.DataFrame:
    df = pd.read_excel(scope_path, sheet_name=sheet_name, engine="openpyxl") if sheet_name else pd.read_excel(scope_path, engine="openpyxl")
    if "Server" not in df.columns:
        raise ValueError('Scoped server file must have a column named "Server".')
    df["Server"] = df["Server"].astype(str).str.strip()
    df["Server_UPPER"] = df["Server"].str.upper()
    return df

def run_premigration_pipeline(rvtools_path: str, scope_path: str, out_path: str, scope_sheet: Optional[str] = None, log_func=print) -> Dict[str, Any]:
    rvtools_path_p = Path(rvtools_path)
    scope_path_p = Path(scope_path)
    out_path_p = Path(out_path)

    log_func(f"[PRE] Loading RVTools export: {rvtools_path_p}")
    dfs = load_rvtools(rvtools_path_p)
    vm_networks = extract_vm_networks_from_rvtools(dfs)

    summary_all, disks, free, iso, snaps = run_rvtools_checks(dfs)
    summary_all = add_ready_and_reasons(summary_all)

    log_func(f"[PRE] Loading scoped server list: {scope_path_p}")
    scoped = load_scoped(scope_path_p, sheet_name=scope_sheet)

    summary_reset = summary_all.reset_index().rename(columns={"index": "VM"})
    summary_reset["VM"] = summary_reset["VM"].astype(str).str.strip()
    summary_reset["VM_UPPER"] = summary_reset["VM"].str.upper()

    merged = scoped.merge(summary_reset, left_on="Server_UPPER", right_on="VM_UPPER", how="left", suffixes=("_scope", "_summary"))
    scoped_found = merged[~merged["VM"].isna()].copy()
    scoped_not_found = merged[merged["VM"].isna()][["Server"]].drop_duplicates()

    scoped_vm_set = set(scoped_found["VM"].astype(str).str.strip())

    disks_scoped = [(vm, issues) for vm, issues in disks.items() if vm in scoped_vm_set and vm != "_error"]
    free_scoped = [(vm, issues) for vm, issues in free.items() if vm in scoped_vm_set]
    iso_scoped = [(vm, True) for vm in iso if vm in scoped_vm_set]
    snaps_scoped = [(vm, issues) for vm, issues in snaps.items() if vm in scoped_vm_set]

    scoped_found.drop(columns=["Server_UPPER", "VM_UPPER"], errors="ignore", inplace=True)

    base_pairs = scoped_found[["Server", "VM"]].drop_duplicates()
    net_rows = []
    for _, row in base_pairs.iterrows():
        server = row["Server"]
        vm = str(row["VM"]).strip()
        nets = vm_networks.get(vm, [])
        net_rows.append({"Server": server, "VM": vm, "NetworkCount": len(nets), "Networks": ", ".join(nets)})
    scoped_networks_df = pd.DataFrame(net_rows, columns=["Server", "VM", "NetworkCount", "Networks"])

    log_func(f"[PRE] Writing scoped pre-migration report: {out_path_p}")
    with pd.ExcelWriter(out_path_p, engine="openpyxl") as writer:
        scoped_found.to_excel(writer, sheet_name="ScopedSummary", index=False)
        scoped_not_found.to_excel(writer, sheet_name="NotFound", index=False)
        pd.DataFrame(disks_scoped, columns=["VM", "DiskIssues"]).to_excel(writer, sheet_name="DiskNotAligned_Scoped", index=False)
        pd.DataFrame(free_scoped, columns=["VM", "LowSpace"]).to_excel(writer, sheet_name="LowFreeSpace_Scoped", index=False)
        pd.DataFrame(iso_scoped, columns=["VM", "HasDevice"]).to_excel(writer, sheet_name="ISO_Devices_Scoped", index=False)
        pd.DataFrame(snaps_scoped, columns=["VM", "Snapshots"]).to_excel(writer, sheet_name="Snapshots_Scoped", index=False)
        scoped_networks_df.to_excel(writer, sheet_name="ScopedNetworks", index=False)

    found_count = scoped_found["Server"].nunique() if "Server" in scoped_found.columns else len(scoped_found)
    log_func("[PRE] === Pre-Migration Health Check DONE ===")
    log_func(f"[PRE] Scoped servers total : {len(scoped)}")
    log_func(f"[PRE] Found in RVTools     : {found_count}")
    log_func(f"[PRE] Not found           : {len(scoped_not_found)}")
    log_func(f"[PRE] Output file         : {out_path_p}")

    return {"output_excel": str(out_path_p), "scoped_found": int(len(scoped_found)), "scoped_not_found": int(len(scoped_not_found))}

def _build_vm_resources_from_rvtools(dfs: Dict[str, pd.DataFrame], log_func=print):
    vcpu_df = dfs.get("vCPU")
    vmem_df = dfs.get("vMemory")
    vdisk_df = dfs.get("vDisk")

    if vcpu_df is None or vcpu_df.empty:
        raise ValueError("RVTools missing 'vCPU' sheet. Cannot calculate CPU totals.")
    if vmem_df is None or vmem_df.empty:
        raise ValueError("RVTools missing 'vMemory' sheet. Cannot calculate memory totals.")
    if vdisk_df is None or vdisk_df.empty:
        raise ValueError("RVTools missing 'vDisk' sheet. Cannot calculate storage totals.")

    vcpu_df = vcpu_df.copy()
    vmem_df = vmem_df.copy()
    vdisk_df = vdisk_df.copy()

    vcpu_df.columns = [str(c).strip() for c in vcpu_df.columns]
    vmem_df.columns = [str(c).strip() for c in vmem_df.columns]
    vdisk_df.columns = [str(c).strip() for c in vdisk_df.columns]

    vm_col_cpu = find_col(vcpu_df, ["vm", "name"])
    cpu_col = find_col(vcpu_df, ["cpu", "cpus"])
    vm_col_mem = find_col(vmem_df, ["vm", "name"])
    mem_mib_col = find_col(vmem_df, ["size mib", "sizemib"])
    vm_col_disk = find_col(vdisk_df, ["vm", "name"])
    cap_mib_col = find_col(vdisk_df, ["capacity mib", "cap mib"])

    if not all([vm_col_cpu, cpu_col, vm_col_mem, mem_mib_col, vm_col_disk, cap_mib_col]):
        raise ValueError("RVTools columns are missing in vCPU/vMemory/vDisk. Cannot run assessment.")

    vcpu_df["VM"] = vcpu_df[vm_col_cpu].astype(str).str.strip()
    vcpu_df["CPUs"] = pd.to_numeric(vcpu_df[cpu_col], errors="coerce").fillna(0.0)
    cpu_group = vcpu_df.groupby("VM")["CPUs"].sum().reset_index()

    vmem_df["VM"] = vmem_df[vm_col_mem].astype(str).str.strip()
    vmem_df["MemGiB"] = pd.to_numeric(vmem_df[mem_mib_col], errors="coerce").fillna(0.0) / 1024.0
    mem_group = vmem_df.groupby("VM")["MemGiB"].sum().reset_index()

    vdisk_df["VM"] = vdisk_df[vm_col_disk].astype(str).str.strip()
    vdisk_df["ProvTB"] = pd.to_numeric(vdisk_df[cap_mib_col], errors="coerce").fillna(0.0) / (1024.0 * 1024.0)

    used_mib_col = find_col(vdisk_df, ["used mib", "inuse mib", "used mib (total)", "usedmb", "inuse mib (total)"])
    if used_mib_col:
        vdisk_df["UsedTB"] = pd.to_numeric(vdisk_df[used_mib_col], errors="coerce").fillna(0.0) / (1024.0 * 1024.0)
    else:
        vdisk_df["UsedTB"] = 0.0

    disk_group = vdisk_df.groupby("VM")[["ProvTB", "UsedTB"]].sum().reset_index()

    df_vms = cpu_group.merge(mem_group, on="VM", how="outer").merge(disk_group, on="VM", how="outer").fillna(0.0)
    return df_vms, vdisk_df

def _calculate_for_node_type(total_vcpu: float, total_mem_gib: float, total_stor_tb: float, node_type_name: str, profile: Dict[str, float]):
    cores = float(profile["cores"])
    mem_gib = float(profile["mem_gib"])
    raw_tb = float(profile["raw_tb"])

    cpu_overcommit = 4.0
    cpu_headroom = 20.0
    esxi_overhead = 10.0
    mem_headroom = 20.0
    vsan_slack = 25.0

    node_cpu_capacity = cores * cpu_overcommit
    node_cpu_eff = node_cpu_capacity * (1 - cpu_headroom / 100.0)
    node_mem_eff = mem_gib * (1 - esxi_overhead / 100.0) * (1 - mem_headroom / 100.0)

    vsan_eff_r1 = 0.50
    vsan_eff_r5 = 0.75

    def nodes_for_storage(total_tb: float, per_node_raw: float, eff: float, slack: float) -> int:
        usable = per_node_raw * eff * (1 - slack / 100.0)
        if usable <= 0:
            return 0
        return max(1, math.ceil(total_tb / usable))

    nodes_cpu = max(3, math.ceil(total_vcpu / node_cpu_eff)) if node_cpu_eff > 0 else 3
    nodes_mem = max(3, math.ceil(total_mem_gib / node_mem_eff)) if node_mem_eff > 0 else 3
    nodes_stor_r1 = max(3, nodes_for_storage(total_stor_tb, raw_tb, vsan_eff_r1, vsan_slack))
    nodes_stor_r5 = max(4, nodes_for_storage(total_stor_tb, raw_tb, vsan_eff_r5, vsan_slack))

    def limiting(cpu: int, mem: int, stor: int) -> str:
        if cpu >= mem and cpu >= stor:
            return "CPU"
        if mem >= cpu and mem >= stor:
            return "Memory"
        return "Storage"

    rec_r1 = max(nodes_cpu, nodes_mem, nodes_stor_r1)
    rec_r5 = max(nodes_cpu, nodes_mem, nodes_stor_r5)

    result_r1 = dict(NodeType=node_type_name, Scenario="FTT1-RAID1 (Mirroring)", Total_vCPU=round(total_vcpu, 2), Total_Mem_GiB=round(total_mem_gib, 2),
                     Total_Storage_TB=round(total_stor_tb, 2), Nodes_by_CPU=nodes_cpu, Nodes_by_Memory=nodes_mem, Nodes_by_Storage=nodes_stor_r1,
                     Recommended_Nodes=rec_r1, Limiting_Factor=limiting(nodes_cpu, nodes_mem, nodes_stor_r1))
    result_r5 = dict(NodeType=node_type_name, Scenario="FTT1-RAID5 (Erasure Coding)", Total_vCPU=round(total_vcpu, 2), Total_Mem_GiB=round(total_mem_gib, 2),
                     Total_Storage_TB=round(total_stor_tb, 2), Nodes_by_CPU=nodes_cpu, Nodes_by_Memory=nodes_mem, Nodes_by_Storage=nodes_stor_r5,
                     Recommended_Nodes=rec_r5, Limiting_Factor=limiting(nodes_cpu, nodes_mem, nodes_stor_r5))
    return result_r1, result_r5

def _calculate_and_write_assessment(df_vms: pd.DataFrame, vdisk_df: pd.DataFrame, total_vcpu: float, total_mem_gib: float, total_stor_tb: float, out_path: Path, mode_label: str, log_func=print):
    summary_rows = []
    log_func(f"[ASSM] ({mode_label}) In-scope VMs: {len(df_vms)}")
    log_func(f"[ASSM] ({mode_label}) Total vCPU     = {total_vcpu:.2f}")
    log_func(f"[ASSM] ({mode_label}) Total Memory   = {total_mem_gib:.2f} GiB")
    log_func(f"[ASSM] ({mode_label}) Total Storage  = {total_stor_tb:.2f} TB")

    for node_type_name, profile in AVS_NODE_TYPES.items():
        r1, r5 = _calculate_for_node_type(total_vcpu, total_mem_gib, total_stor_tb, node_type_name, profile)
        r1["Scenario"] = f"{r1['Scenario']} – {mode_label}"
        r5["Scenario"] = f"{r5['Scenario']} – {mode_label}"
        summary_rows += [r1, r5]

    df_summary = pd.DataFrame(summary_rows)
    log_func(f"[ASSM] ({mode_label}) Writing AVS capacity assessment to: {out_path}")
    with pd.ExcelWriter(out_path, engine="openpyxl") as writer:
        df_summary.to_excel(writer, index=False, sheet_name="Summary")
        df_vms.to_excel(writer, index=False, sheet_name="PerVM")
        vdisk_df.to_excel(writer, index=False, sheet_name="vDisksRaw")
    return {"output_excel": str(out_path), "summary_rows": int(len(df_summary)), "pervm_rows": int(len(df_vms))}

def run_assessment_pipeline_usedpreferred(rvtools_path: str, out_path: str, log_func=print):
    dfs = load_rvtools(Path(rvtools_path))
    df_vms, vdisk_df = _build_vm_resources_from_rvtools(dfs, log_func=log_func)
    df_vms["SizingTB"] = df_vms.apply(lambda r: r["UsedTB"] if r["UsedTB"] > 0 else r["ProvTB"], axis=1)
    total_vcpu = float(df_vms["CPUs"].sum())
    total_mem_gib = float(df_vms["MemGiB"].sum())
    total_stor_tb = float(df_vms["SizingTB"].sum())
    return _calculate_and_write_assessment(df_vms, vdisk_df, total_vcpu, total_mem_gib, total_stor_tb, Path(out_path), "UsedPreferred", log_func=log_func)

def run_assessment_pipeline_provisioned(rvtools_path: str, out_path: str, log_func=print):

def _build_vm_cpu_mem_from_rvtools(dfs: Dict[str, pd.DataFrame]) -> pd.DataFrame:
    vcpu_df = dfs.get("vCPU")
    vmem_df = dfs.get("vMemory")
    if vcpu_df is None or vcpu_df.empty:
        raise ValueError("RVTools missing 'vCPU' sheet.")
    if vmem_df is None or vmem_df.empty:
        raise ValueError("RVTools missing 'vMemory' sheet.")

    vcpu_df = vcpu_df.copy()
    vmem_df = vmem_df.copy()
    vcpu_df.columns = [str(c).strip() for c in vcpu_df.columns]
    vmem_df.columns = [str(c).strip() for c in vmem_df.columns]

    vm_col_cpu = find_col(vcpu_df, ["vm", "name"])
    cpu_col = find_col(vcpu_df, ["cpu", "cpus"])
    vm_col_mem = find_col(vmem_df, ["vm", "name"])
    mem_mib_col = find_col(vmem_df, ["size mib", "sizemib"])
    if not all([vm_col_cpu, cpu_col, vm_col_mem, mem_mib_col]):
        raise ValueError("RVTools columns missing in vCPU/vMemory.")

    vcpu_df["VM"] = vcpu_df[vm_col_cpu].astype(str).str.strip()
    vcpu_df["vCPU"] = pd.to_numeric(vcpu_df[cpu_col], errors="coerce").fillna(0.0)

    vmem_df["VM"] = vmem_df[vm_col_mem].astype(str).str.strip()
    vmem_df["MemGiB"] = pd.to_numeric(vmem_df[mem_mib_col], errors="coerce").fillna(0.0) / 1024.0

    cpu_group = vcpu_df.groupby("VM")["vCPU"].sum().reset_index()
    mem_group = vmem_df.groupby("VM")["MemGiB"].sum().reset_index()
    return cpu_group.merge(mem_group, on="VM", how="outer").fillna(0.0)

def _recommend_azure_size(req_vcpu: float, req_mem_gib: float, allowed_families: set) -> Optional[Dict[str, Any]]:
    req_vcpu_int = int(math.ceil(req_vcpu))
    candidates = [s for s in AZURE_SIZE_CATALOG if s["family"] in allowed_families and s["vcpus"] >= req_vcpu_int and s["mem_gib"] >= float(req_mem_gib)]
    if not candidates:
        return None
    candidates.sort(key=lambda x: (x["vcpus"], x["mem_gib"]))
    return candidates[0]

def run_azure_vm_assessment_pipeline(rvtools_path: str, out_path: str, family_mode: str = "AUTO", headroom_pct: float = 20.0, log_func=print) -> Dict[str, Any]:
    if headroom_pct < 0:
        raise ValueError("Headroom % must be >= 0")
    allowed = {"Dsv5", "Esv5"} if family_mode.upper() == "AUTO" else ({"Dsv5"} if family_mode.upper() == "DSV5" else {"Esv5"})

    dfs = load_rvtools(Path(rvtools_path))
    df = _build_vm_cpu_mem_from_rvtools(dfs)
    if df.empty:
        raise ValueError("No VMs found in vCPU/vMemory sheets.")

    df["HeadroomPct"] = headroom_pct
    df["Req_vCPU"] = df["vCPU"] * (1.0 + headroom_pct / 100.0)
    df["Req_MemGiB"] = df["MemGiB"] * (1.0 + headroom_pct / 100.0)

    rec_names, rec_fams, rec_vcpus, rec_mem, notes = [], [], [], [], []
    for _, r in df.iterrows():
        rec = _recommend_azure_size(r["Req_vCPU"], r["Req_MemGiB"], allowed)
        if rec is None:
            rec_names.append("")
            rec_fams.append("")
            rec_vcpus.append(None)
            rec_mem.append(None)
            notes.append("NO_MATCH_IN_CATALOG (extend AZURE_SIZE_CATALOG)")
        else:
            rec_names.append(rec["name"])
            rec_fams.append(rec["family"])
            rec_vcpus.append(rec["vcpus"])
            rec_mem.append(rec["mem_gib"])
            notes.append("")

    df["RecommendedSize"] = rec_names
    df["RecommendedFamily"] = rec_fams
    df["Recommended_vCPU"] = rec_vcpus
    df["Recommended_MemGiB"] = rec_mem
    df["Notes"] = notes

    summary = (df[df["RecommendedSize"].astype(str).str.strip() != ""]
               .groupby(["RecommendedFamily", "RecommendedSize"]).size()
               .reset_index(name="VMCount")
               .sort_values(["RecommendedFamily", "VMCount"], ascending=[True, False]))
    not_mapped = df[df["RecommendedSize"].astype(str).str.strip() == ""].copy()
    catalog_df = pd.DataFrame(AZURE_SIZE_CATALOG)

    out_path_p = Path(out_path)
    log_func(f"[AZURE] Writing Azure VM assessment to: {out_path_p}")
    with pd.ExcelWriter(out_path_p, engine="openpyxl") as writer:
        summary.to_excel(writer, sheet_name="Summary", index=False)
        df.to_excel(writer, sheet_name="PerVM", index=False)
        not_mapped.to_excel(writer, sheet_name="NotMapped", index=False)
        catalog_df.to_excel(writer, sheet_name="CatalogUsed", index=False)

    return {"output_excel": str(out_path_p), "vms_total": int(len(df)), "vms_sized": int(len(df) - len(not_mapped)), "vms_not_mapped": int(len(not_mapped))}

def run_ping(server: str, count: int = 2, timeout: int = 3):
    if os.name == "nt":
        cmd = ["ping", "-n", str(count), "-w", str(timeout * 1000), server]
    else:
        cmd = ["ping", "-c", str(count), "-W", str(timeout), server]
    try:
        completed = subprocess.run(cmd, capture_output=True, text=True, timeout=timeout * count + 5)
        success = (completed.returncode == 0)
        output = (completed.stdout or "") + "\n" + (completed.stderr or "")
    except Exception as e:
        success = False
        output = f"ERROR running ping: {e}"
    return success, output

def run_traceroute(server: str, max_hops: int = 20, timeout: int = 60):
    if os.name == "nt":
        cmd = ["tracert", "-d", "-h", str(max_hops), server]
    else:
        cmd = ["traceroute", "-n", "-m", str(max_hops), server]
    try:
        completed = subprocess.run(cmd, capture_output=True, text=True, timeout=timeout)
        success = (completed.returncode == 0)
        output = (completed.stdout or "") + "\n" + (completed.stderr or "")
    except Exception as e:
        success = False
        output = f"ERROR running traceroute: {e}"
    return success, output

def run_postmigration_pipeline(scope_path: str, out_path: str, log_func=print) -> Dict[str, Any]:
    scoped = load_scoped(Path(scope_path), sheet_name=None)
    servers = sorted(set(scoped["Server"].astype(str).str.strip()))
    servers = [s for s in servers if s]

    rows = []
    for server in servers:
        log_func(f"[POST] Pinging {server} ...")
        ping_ok, ping_out = run_ping(server)
        log_func(f"[POST] Traceroute {server} ...")
        trace_ok, trace_out = run_traceroute(server)
        rows.append({"Server": server, "PingSuccess": ping_ok, "TracerouteSuccess": trace_ok, "PingOutput": ping_out, "TracerouteOutput": trace_out})

    df = pd.DataFrame(rows)
    out_path_p = Path(out_path)
    out_path_p.parent.mkdir(parents=True, exist_ok=True)
    with pd.ExcelWriter(out_path_p, engine="openpyxl") as writer:
        df.to_excel(writer, sheet_name="PostMigrationStatus", index=False)

    success_count = sum(1 for r in rows if r["PingSuccess"])
    return {"output_excel": str(out_path_p), "servers_total": int(len(servers)), "ping_success": int(success_count)}

    # ------------------------------
    # Queue / logging
    # ------------------------------
    def _enqueue_log(self, msg: str):
        self.msg_queue.put(("log", msg))

    def _enqueue_call(self, func, *args, **kwargs):
        self.msg_queue.put(("call", (func, args, kwargs)))

    def _drain_queue(self):
        try:
            while True:
                kind, payload = self.msg_queue.get_nowait()
                if kind == "log":
                    self._log_to_text(payload)
                elif kind == "call":
                    func, args, kwargs = payload
                    func(*args, **kwargs)
        except queue.Empty:
            pass
        self.after(100, self._drain_queue)

    def _on_drop_dep(self, event):
        data = event.data
        paths = []
        if data.strip().startswith("{"):
            parts, buf, brace = [], "", False
            for ch in data:
                if ch == "{":
                    brace, buf = True, ""
                elif ch == "}":
                    brace = False
                    if buf:
                        parts.append(buf); buf = ""
                elif brace:
                    buf += ch
                elif ch == " " and not brace:
                    if buf:
                        parts.append(buf); buf = ""
                else:
                    buf += ch
            if buf:
                parts.append(buf)
            for p in parts:
                if os.path.isfile(p):
                    paths.append(p)
        else:
            for token in data.split():
                if os.path.isfile(token):
                    paths.append(token)

        if paths:
            self._set_dep_selected(paths)
        else:
            messagebox.showerror("Invalid file(s)", f"Couldn't read dropped item(s):\n{data}")

    def _run_dep_pipeline_threaded(self):
        paths = list(self.dep_files)
        if not paths:
            raw = self.file_path.get().strip()
            if raw:
                candidates = [p.strip() for p in raw.split(";") if p.strip()]
                paths = [p for p in candidates if os.path.isfile(p)]

            messagebox.showerror("No file", "Please select one or more Azure Migrate dependency files first.")

def find_col(df: pd.DataFrame, patterns: List[str]) -> Optional[str]:
    """Find a column whose name contains any of the patterns (case-insensitive)."""
    for col in df.columns:
        low = str(col).lower()
        for p in patterns:
            if p.lower() in low:
                return col
    return None


def ip_to_24(ip: str) -> Optional[str]:
    ip = str(ip).strip()
    if re.match(r"^\d{1,3}(\.\d{1,3}){3}$", ip):
        a, b, c, _ = ip.split(".")
        return f"{a}.{b}.{c}.0/24"
    return None


def read_any(path: str) -> pd.DataFrame:
    p = Path(path)
    if not p.exists():
        raise FileNotFoundError(path)

    ext = p.suffix.lower()
    if ext == ".csv":
        df = pd.read_csv(p)
    elif ext in (".xlsx", ".xls"):
        df = pd.read_excel(p)
    else:
        raise ValueError(f"Unsupported file type: {ext}. Use .csv/.xlsx/.xls")

    df.columns = [str(c).strip() for c in df.columns]
    return df


def canonicalize(df: pd.DataFrame) -> pd.DataFrame:
    """
    Tries to map your export headers to canonical names:
    Timestamp, RuleId, Action, Direction, Protocol, SourceIp, DestinationIp, SourcePort, DestinationPort
    """
    df = df.copy()

    ts = find_col(df, ["timestamp"]) or find_col(df, ["time"])
    rule = find_col(df, ["ruleid", "rule id", "rule_id"])
    action = find_col(df, ["action"])
    direction = find_col(df, ["direction"])
    proto = find_col(df, ["protocol", "proto"])
    src_ip = find_col(df, ["sourceip", "source ip", "srcip", "src ip"])
    dst_ip = find_col(df, ["destinationip", "destination ip", "dstip", "dst ip", "dest ip"])
    src_port = find_col(df, ["sourceport", "source port", "srcport", "src port"])
    dst_port = find_col(df, ["destinationport", "destination port", "dstport", "dst port", "dport", "dest port"])

    required = {
        "Timestamp": ts,
        "RuleId": rule,
        "Action": action,
        "Direction": direction,
        "Protocol": proto,
        "SourceIp": src_ip,
        "DestinationIp": dst_ip,
        "SourcePort": src_port,
        "DestinationPort": dst_port,
    }
    missing = [k for k, v in required.items() if v is None]
    if missing:
        raise ValueError(f"Missing required columns: {missing}\nFound columns: {list(df.columns)}")

    df = df.rename(columns={
        ts: "Timestamp",
        rule: "RuleId",
        action: "Action",
        direction: "Direction",
        proto: "Protocol",
        src_ip: "SourceIp",
        dst_ip: "DestinationIp",
        src_port: "SourcePort",
        dst_port: "DestinationPort",
    })
    return df

def run_csv_analysis(
    input_files: List[str],
    output_dir: str,
    pass_only: bool = True,
    src_agg_threshold: int = 20,
    log=print
) -> Dict[str, str]:
    out = Path(output_dir).expanduser().resolve()
    out.mkdir(parents=True, exist_ok=True)

    frames = []
    for f in input_files:
        log(f"[NSX] Reading: {f}")
        df = canonicalize(read_any(f))
        df["SourceFile"] = os.path.basename(f)
        frames.append(df)

    df_all = pd.concat(frames, ignore_index=True)

    # Normalize
    df_all["Timestamp"] = pd.to_datetime(df_all["Timestamp"], errors="coerce", utc=True).dt.tz_convert(None)
    for col in ["SourcePort", "DestinationPort"]:
        df_all[col] = pd.to_numeric(df_all[col], errors="coerce").astype("Int64")

    df_all["Protocol"] = df_all["Protocol"].fillna("Any").astype(str).str.strip().str.upper()
    df_all["Action"] = df_all["Action"].fillna("Unknown").astype(str).str.strip().str.upper()
    df_all["Direction"] = df_all["Direction"].fillna("Unknown").astype(str).str.strip()

    # 0) Input summary
    input_summary = (
        df_all.groupby("SourceFile")
        .agg(
            Rows=("Timestamp", "size"),
            TimeMin=("Timestamp", "min"),
            TimeMax=("Timestamp", "max"),
            UniqueRuleIds=("RuleId", pd.Series.nunique),
            UniqueSourceIps=("SourceIp", lambda s: s.dropna().nunique()),
            UniqueDestIps=("DestinationIp", lambda s: s.dropna().nunique()),
        ).reset_index()
    )
    input_summary.to_csv(out / "Input_Summary.csv", index=False)

    # 1) Rule hits
    rule_hits = (
        df_all.groupby(["RuleId", "Protocol", "DestinationPort", "Direction", "Action"])
        .size().reset_index(name="HitCount")
        .sort_values("HitCount", ascending=False)
    )
    rule_hits.to_csv(out / "RuleId_Hits.csv", index=False)

    # 2) Dest/Port analysis
    dest_port = (
        df_all.groupby(["SourceFile", "DestinationIp", "Protocol", "DestinationPort"])
        .agg(
            HitCount=("Timestamp", "size"),
            UniqueSources=("SourceIp", lambda s: s.dropna().nunique()),
            SourceIps=("SourceIp", lambda s: ", ".join(sorted(set(s.dropna().astype(str)))[:50])),
            RuleIds=("RuleId", lambda s: ", ".join(map(str, sorted(set(pd.to_numeric(s, errors="coerce").dropna().astype(int)))))),
        ).reset_index()
        .sort_values("HitCount", ascending=False)
    )
    dest_port.to_csv(out / "Dest_Port_Analysis.csv", index=False)

    # 3) Top flows (Top 2000)
    top_flows = (
        df_all.groupby(["SourceFile", "SourceIp", "DestinationIp", "Protocol", "DestinationPort", "Action", "RuleId"])
        .size().reset_index(name="HitCount")
        .sort_values("HitCount", ascending=False)
        .head(2000)
    )
    top_flows.to_csv(out / "Top_Flows_Top2000.csv", index=False)

    # 4) Proposed NSG rules
    cand = df_all.copy()
    if pass_only:
        cand = cand[cand["Action"] == "PASS"].copy()

    cand = cand.dropna(subset=["DestinationIp", "DestinationPort"])

    grp = (
        cand.groupby(["DestinationIp", "Protocol", "DestinationPort"])
        .agg(
            HitCount=("Timestamp", "size"),
            SourceIps=("SourceIp", lambda s: sorted(set(s.dropna().astype(str)))),
            RuleIds=("RuleId", lambda s: sorted(set(pd.to_numeric(s, errors="coerce").dropna().astype(int)))),
        ).reset_index()
        .sort_values("HitCount", ascending=False)
    )

    rows = []
    prio = 1000
    for _, r in grp.iterrows():
        src_ips = r["SourceIps"]
        src_24 = sorted(set(filter(None, (ip_to_24(ip) for ip in src_ips))))
        use_24 = (len(src_ips) > src_agg_threshold) and (len(src_24) <= len(src_ips)) and (len(src_24) > 0)

        src_repr = ";".join(src_24 if use_24 else src_ips)
        dst_ip = str(r["DestinationIp"]).strip()
        proto = str(r["Protocol"]).strip().upper()
        dport = int(r["DestinationPort"])

        rows.append({
            "Name": f"allow-{dst_ip.replace('.','-')}-{proto.lower()}-{dport}"[:80],
            "Priority": prio,
            "Direction": "Inbound",
            "Access": "Allow",
            "Protocol": proto if proto in ["TCP", "UDP", "ICMP"] else "*",
            "SourceAddressPrefixes": src_repr,
            "SourcePortRanges": "*",
            "DestinationAddressPrefix": f"{dst_ip}/32",
            "DestinationPortRanges": str(dport),
            "Description": f"From NSX logs; RuleIds={','.join(map(str, r['RuleIds']))}; Hits={int(r['HitCount'])}",
        })
        prio += 10

    pd.DataFrame(rows).to_csv(out / "Proposed_NSG_Rules.csv", index=False)

    return {
        "OutputFolder": str(out),
        "Input_Summary": str(out / "Input_Summary.csv"),
        "RuleId_Hits": str(out / "RuleId_Hits.csv"),
        "Dest_Port_Analysis": str(out / "Dest_Port_Analysis.csv"),
        "Top_Flows_Top2000": str(out / "Top_Flows_Top2000.csv"),
        "Proposed_NSG_Rules": str(out / "Proposed_NSG_Rules.csv"),
    }

        self.log = ScrolledText(frm, height=18, state="disabled", wrap="word")
        self.log.pack(fill="both", expand=True, pady=(8, 0))

        self.after(100, self.drain)

    def log_line(self, s: str):
        self.q.put(("log", s))

    def call(self, fn, *a, **k):
        self.q.put(("call", (fn, a, k)))

    def drain(self):
        try:
            while True:
                kind, payload = self.q.get_nowait()
                if kind == "log":
                    self.log.configure(state="normal")
                    self.log.insert("end", payload + "\n")
                    self.log.see("end")
                    self.log.configure(state="disabled")
                else:
                    fn, a, k = payload
                    fn(*a, **k)
        except queue.Empty:
            pass
        self.after(100, self.drain)

    def browse(self):
        fps = filedialog.askopenfilenames(
            title="Select NSX-T exports",
            filetypes=[("CSV/Excel", "*.csv *.xlsx *.xls"), ("All files", "*.*")]
        )
        if fps:
            self.files = list(fps)
            self.files_var.set(" ; ".join(self.files))

    def clear(self):
        self.files = []
        self.files_var.set("")

    def browse_out(self):
        d = filedialog.askdirectory(title="Select output folder")
        if d:
            self.out_var.set(d)

    def run_threaded(self):
        files = self.files or [p.strip() for p in self.files_var.get().split(";") if p.strip()]
        files = [p for p in files if os.path.isfile(p)]
        if not files:
            messagebox.showerror("Inputs", "Select at least one input file.")
            return

        out_base = self.out_var.get().strip()
        if not out_base:
            messagebox.showerror("Output", "Select an output folder.")
            return

        try:
            thr = int(float(self.threshold_var.get().strip()))
        except Exception:
            messagebox.showerror("Threshold", "Threshold must be a number (e.g., 20).")
            return

        run_dir = str(Path(out_base) / f"NSXT_to_NSG_CSV_{datetime.now().strftime('%Y%m%d_%H%M%S')}")
        pass_only = bool(self.pass_only_var.get())

        self.run_btn.configure(state="disabled")
        self.pb.start(12)

        self.log_line(f"[NSX] Output: {run_dir}")
        for f in files:
            self.log_line(f"[NSX] Input : {f}")

        def work():
            try:
                outputs = run_csv_analysis(files, run_dir, pass_only, thr, log=self.log_line)
                self.call(self.done, outputs, run_dir)
            except Exception as e:
                self.call(self.fail, str(e))

        threading.Thread(target=work, daemon=True).start()

    def done(self, outputs: Dict[str, str], run_dir: str):
        self.pb.stop()
        self.run_btn.configure(state="normal")
        self.log_line("[NSX] Done. Files:")
        for k, v in outputs.items():
            self.log_line(f"  - {k}: {v}")
        messagebox.showinfo("Done", f"Completed.\n\n{run_dir}")

    def fail(self, msg: str):
        self.pb.stop()
        self.run_btn.configure(state="normal")
        self.log_line(f"[NSX] ERROR: {msg}")
        messagebox.showerror("Error", msg)

# ==========================================================
# Shared helpers

def _normalize_text(s: Any) -> str:
    return str(s).strip()

def _normalize_port(p: Any) -> Optional[str]:
    s = str(p).strip()
    if not s or s.lower() in {"nan", "none", "n/a"}:
        return None
    if s.endswith(".0"):
        s = s[:-2]
    return s

def find_col(df: pd.DataFrame, patterns: List[str]) -> Optional[str]:
    """Find a column whose name contains any of the patterns (case-insensitive)."""
    for col in df.columns:
        low = str(col).lower()
        for p in patterns:
            if p.lower() in low:
                return col
    return None

def _split_multi(s: str) -> List[str]:
    return [x.strip() for x in re.split(r"[,;\s]+", s or "") if x.strip()]

def _read_any_table(path: str) -> pd.DataFrame:
    p = Path(path)
    if not p.exists():
        raise FileNotFoundError(f"File not found: {p}")
    ext = p.suffix.lower()
    if ext == ".csv":
        return pd.read_csv(p)
    if ext in (".xlsx", ".xls"):
        return pd.read_excel(p)
    raise ValueError("Unsupported mapping file type. Use CSV or Excel.")

def _normalize_hostname_for_matching(name: Any) -> str:
    """Normalize hostname / asset tag for CMDB matching."""
    s = str(name).strip().upper()
    if not s:
        return ""
    if "." in s:
        s = s.split(".", 1)[0]
    return s

def _run_embedded_powershell(script_text: str, args: List[str], log_func=print) -> int:
    """
    Write script_text to a temp .ps1, execute PowerShell with args, delete temp file.
    Returns exit code.
    """
    if os.name != "nt":
        raise RuntimeError("PowerShell embedded scripts require Windows.")
    with tempfile.NamedTemporaryFile(delete=False, suffix=".ps1", mode="w", encoding="utf-8") as tmp:
        tmp.write(script_text)
        ps1_path = tmp.name
    cmd = ["powershell.exe", "-NoProfile", "-ExecutionPolicy", "Bypass", "-File", ps1_path] + args
    log_func(f"[PS] Running: {' '.join(cmd[:6])} ... (args hidden)")
    try:
        proc = subprocess.Popen(cmd)
        return proc.wait()
    finally:
        try:
            os.remove(ps1_path)
        except Exception:
            pass

def _nsx_ip_to_24(ip: str) -> Optional[str]:
    ip = str(ip).strip()
    if re.match(r"^\d{1,3}(\.\d{1,3}){3}$", ip):
        a, b, c, _ = ip.split(".")
        return f"{a}.{b}.{c}.0/24"
    return None

def _nsx_canonicalize(df: pd.DataFrame) -> pd.DataFrame:
    """
    Map export headers to canonical names:
    Timestamp, RuleId, Action, Direction, Protocol, SourceIp, DestinationIp, SourcePort, DestinationPort
    """
    df = df.copy()
    ts = find_col(df, ["timestamp"]) or find_col(df, ["time"])
    rule = find_col(df, ["ruleid", "rule id", "rule_id"])
    action = find_col(df, ["action"])
    direction = find_col(df, ["direction"])
    proto = find_col(df, ["protocol", "proto"])
    src_ip = find_col(df, ["sourceip", "source ip", "srcip", "src ip"])
    dst_ip = find_col(df, ["destinationip", "destination ip", "dstip", "dst ip", "dest ip"])
    src_port = find_col(df, ["sourceport", "source port", "srcport", "src port"])
    dst_port = find_col(df, ["destinationport", "destination port", "dstport", "dst port", "dport", "dest port"])

    required = {
        "Timestamp": ts,
        "RuleId": rule,
        "Action": action,
        "Direction": direction,
        "Protocol": proto,
        "SourceIp": src_ip,
        "DestinationIp": dst_ip,
        "SourcePort": src_port,
        "DestinationPort": dst_port,
    }
    missing = [k for k, v in required.items() if v is None]
    if missing:
        raise ValueError(f"Missing required columns: {missing}\nFound columns: {list(df.columns)}")

    df = df.rename(columns={
        ts: "Timestamp",
        rule: "RuleId",
        action: "Action",
        direction: "Direction",
        proto: "Protocol",
        src_ip: "SourceIp",
        dst_ip: "DestinationIp",
        src_port: "SourcePort",
        dst_port: "DestinationPort",
    })
    return df

def run_nsx_csv_analysis(
    input_files: List[str],
    output_dir: str,
    pass_only: bool = True,
    src_agg_threshold: int = 20,
    log=print
) -> Dict[str, str]:
    out = Path(output_dir).expanduser().resolve()
    out.mkdir(parents=True, exist_ok=True)
    frames = []
    for f in input_files:
        log(f"[NSX] Reading: {f}")
        raw_df = _read_any_table(f)
        raw_df.columns = [str(c).strip() for c in raw_df.columns]
        df = _nsx_canonicalize(raw_df)
        df["SourceFile"] = os.path.basename(f)
        frames.append(df)
    df_all = pd.concat(frames, ignore_index=True)

    # Normalize
    df_all["Timestamp"] = pd.to_datetime(df_all["Timestamp"], errors="coerce", utc=True).dt.tz_convert(None)
    for col in ["SourcePort", "DestinationPort"]:
        df_all[col] = pd.to_numeric(df_all[col], errors="coerce").astype("Int64")
    df_all["Protocol"] = df_all["Protocol"].fillna("Any").astype(str).str.strip().str.upper()
    df_all["Action"] = df_all["Action"].fillna("Unknown").astype(str).str.strip().str.upper()
    df_all["Direction"] = df_all["Direction"].fillna("Unknown").astype(str).str.strip()

    # 0) Input summary
    input_summary = (
        df_all.groupby("SourceFile")
        .agg(
            Rows=("Timestamp", "size"),
            TimeMin=("Timestamp", "min"),
            TimeMax=("Timestamp", "max"),
            UniqueRuleIds=("RuleId", pd.Series.nunique),
            UniqueSourceIps=("SourceIp", lambda s: s.dropna().nunique()),
            UniqueDestIps=("DestinationIp", lambda s: s.dropna().nunique()),
        ).reset_index()
    )
    input_summary.to_csv(out / "Input_Summary.csv", index=False)

    # 1) Rule hits
    rule_hits = (
        df_all.groupby(["RuleId", "Protocol", "DestinationPort", "Direction", "Action"])
        .size().reset_index(name="HitCount")
        .sort_values("HitCount", ascending=False)
    )
    rule_hits.to_csv(out / "RuleId_Hits.csv", index=False)

    # 2) Dest/Port analysis
    dest_port = (
        df_all.groupby(["SourceFile", "DestinationIp", "Protocol", "DestinationPort"])
        .agg(
            HitCount=("Timestamp", "size"),
            UniqueSources=("SourceIp", lambda s: s.dropna().nunique()),
            SourceIps=("SourceIp", lambda s: ", ".join(sorted(set(s.dropna().astype(str)))[:50])),
            RuleIds=("RuleId", lambda s: ", ".join(map(str, sorted(set(pd.to_numeric(s, errors="coerce").dropna().astype(int)))))),
        ).reset_index()
        .sort_values("HitCount", ascending=False)
    )
    dest_port.to_csv(out / "Dest_Port_Analysis.csv", index=False)

    # 3) Top flows (Top 2000)
    top_flows = (
        df_all.groupby(["SourceFile", "SourceIp", "DestinationIp", "Protocol", "DestinationPort", "Action", "RuleId"])
        .size().reset_index(name="HitCount")
        .sort_values("HitCount", ascending=False)
        .head(2000)
    )
    top_flows.to_csv(out / "Top_Flows_Top2000.csv", index=False)

    # 4) Proposed NSG rules
    cand = df_all.copy()
    if pass_only:
        cand = cand[cand["Action"] == "PASS"].copy()
    cand = cand.dropna(subset=["DestinationIp", "DestinationPort"])
    grp = (
        cand.groupby(["DestinationIp", "Protocol", "DestinationPort"])
        .agg(
            HitCount=("Timestamp", "size"),
            SourceIps=("SourceIp", lambda s: sorted(set(s.dropna().astype(str)))),
            RuleIds=("RuleId", lambda s: sorted(set(pd.to_numeric(s, errors="coerce").dropna().astype(int)))),
        ).reset_index()
        .sort_values("HitCount", ascending=False)
    )
    rows = []
    prio = 1000
    for _, r in grp.iterrows():
        src_ips = r["SourceIps"]
        src_24 = sorted(set(filter(None, (_nsx_ip_to_24(ip) for ip in src_ips))))
        use_24 = (len(src_ips) > src_agg_threshold) and (len(src_24) <= len(src_ips)) and (len(src_24) > 0)
        src_repr = ";".join(src_24 if use_24 else src_ips)
        dst_ip = str(r["DestinationIp"]).strip()
        proto = str(r["Protocol"]).strip().upper()
        dport = int(r["DestinationPort"])
        rows.append({
            "Name": f"allow-{dst_ip.replace('.','-')}-{proto.lower()}-{dport}"[:80],
            "Priority": prio,
            "Direction": "Inbound",
            "Access": "Allow",
            "Protocol": proto if proto in ["TCP", "UDP", "ICMP"] else "*",
            "SourceAddressPrefixes": src_repr,
            "SourcePortRanges": "*",
            "DestinationAddressPrefix": f"{dst_ip}/32",
            "DestinationPortRanges": str(dport),
            "Description": f"From NSX logs; RuleIds={','.join(map(str, r['RuleIds']))}; Hits={int(r['HitCount'])}",
        })
        prio += 10
    pd.DataFrame(rows).to_csv(out / "Proposed_NSG_Rules.csv", index=False)

    return {
        "OutputFolder": str(out),
        "Input_Summary": str(out / "Input_Summary.csv"),
        "RuleId_Hits": str(out / "RuleId_Hits.csv"),
        "Dest_Port_Analysis": str(out / "Dest_Port_Analysis.csv"),
        "Top_Flows_Top2000": str(out / "Top_Flows_Top2000.csv"),
        "Proposed_NSG_Rules": str(out / "Proposed_NSG_Rules.csv"),
    }

def enrich_ne_status_with_vlan_mapping(
    ne_raw_csv: str,
    vlan_mapping_path: str,
    out_excel_path: str,
    log_func=print
) -> Dict[str, str]:
    """Merge NE status CSV with a CE-NE-VLAN mapping and write a single Excel."""
    status_df = pd.read_csv(ne_raw_csv)
    map_df = _read_any_table(vlan_mapping_path)
    # Heuristics for mapping columns
    name_col = None
    for c in map_df.columns:
        if str(c).strip().lower() == "name":
            name_col = c
            break
    if name_col is None:
        for c in map_df.columns:
            lc = str(c).lower()
            if "network" in lc or "vlan" in lc:
                name_col = c
                break
    ne_id_col = None
    for c in map_df.columns:
        if str(c).strip().lower() in {"ne_id", "ne", "network_extension", "network extension"}:
            ne_id_col = c
            break
    if ne_id_col is None:
        for c in map_df.columns:
            if "ne" in str(c).lower():
                ne_id_col = c
                break
    if name_col is None or ne_id_col is None:
        raise ValueError(f"Could not detect Name/NE columns in mapping. Columns: {list(map_df.columns)}")
    # Normalize join keys by network name
    status_df["__key"] = status_df.get("NetworkName", "").astype(str).str.strip().str.upper()
    map_df["__key"] = map_df[name_col].astype(str).str.strip().str.upper()
    # Optional placeholder drop
    if "IsPlaceholder" in map_df.columns:
        try:
            map_df = map_df[~map_df["IsPlaceholder"].fillna(False).astype(bool)].copy()
        except Exception:
            pass
    merged = map_df.merge(status_df, on="__key", how="left", suffixes=("_MAP", "_NE")).drop(columns=["__key"], errors="ignore")
    # Summary by NE + State
    if "State" in merged.columns:
        summary = (
            merged.groupby(ne_id_col)["State"]
            .value_counts(dropna=False)
            .unstack(fill_value=0)
            .reset_index()
        )
    else:
        summary = merged.groupby(ne_id_col).size().reset_index(name="RowCount")
    log_func(f"[NE] Writing enriched Excel: {out_excel_path}")
    with pd.ExcelWriter(out_excel_path, engine="openpyxl") as writer:
        status_df.drop(columns=["__key"], errors="ignore").to_excel(writer, sheet_name="NE_Status_Raw", index=False)
        map_df.drop(columns=["__key"], errors="ignore").to_excel(writer, sheet_name="VLAN_Mapping", index=False)
        merged.to_excel(writer, sheet_name="NE_VLAN_Status", index=False)
        summary.to_excel(writer, sheet_name="Summary", index=False)
    return {"enriched_excel": out_excel_path}

def _read_dependency_export(input_path: str) -> pd.DataFrame:
    _, ext = os.path.splitext(input_path.lower())
    if ext == ".csv":
        df = pd.read_csv(input_path)
    elif ext in (".xlsx", ".xls"):
        df = pd.read_excel(input_path)
    else:
        raise ValueError("Unsupported file type. Select a CSV or Excel file.")
    df.columns = [str(c).strip() for c in df.columns]
    if all(c in df.columns for c in REQUIRED_COLUMNS):
        src_col, dst_col, port_col = REQUIRED_COLUMNS
    else:
        def _find_cols(patterns):
            pats = [p.lower() for p in patterns]
            matches = []
            for col in df.columns:
                low = str(col).lower()
                if any(p in low for p in pats):
                    matches.append(col)
            return matches

        def _pick_best(pattern_groups, banned_terms=None):
            banned_terms = [b.lower() for b in (banned_terms or [])]
            for group in pattern_groups:
                if isinstance(group, str):
                    group = [group]
                matches = _find_cols(group)
                if not matches:
                    continue
                if banned_terms and len(matches) > 1:
                    filtered = [c for c in matches if not any(b in str(c).lower() for b in banned_terms)]
                    if filtered:
                        matches = filtered
                for p in group:
                    exact = [c for c in matches if str(c).lower().strip() == p.lower().strip()]
                    if exact:
                        return sorted(exact, key=lambda c: len(str(c)))[0]
                return sorted(matches, key=lambda c: len(str(c)))[0]
            return None

        src_col = _pick_best(["source server name", "source machine name", "source host name", "source hostname", "source host", "source machine"], banned_terms=["arm id"])
        dst_col = _pick_best(["destination server name", "destination machine name", "destination host name", "destination hostname", "destination host", "destination machine"], banned_terms=["arm id"])
        port_col = _pick_best(["destination port", "dest port", "remote port", "port"])
        if not (src_col and dst_col and port_col):
            raise ValueError(f"Missing required columns: {REQUIRED_COLUMNS}. Found: {list(df.columns)}")

    df = df.dropna(subset=[src_col, dst_col]).copy()
    df = df.rename(columns={src_col: "Source server name", dst_col: "Destination server name", port_col: "Destination port"})
    df["Source server name"] = df["Source server name"].map(_normalize_text)
    df["Destination server name"] = df["Destination server name"].map(_normalize_text)
    df["Destination port"] = df["Destination port"].map(_normalize_port)
    df = df.dropna(subset=["Destination port"]).copy()
    df["Destination port"] = df["Destination port"].astype(str)
    return df

def _read_dependency_exports(input_paths: Any) -> pd.DataFrame:
    if isinstance(input_paths, (str, Path)):
        paths = [str(input_paths)]
    else:
        paths = [str(p) for p in input_paths]
    if not paths:
        raise ValueError("No dependency files provided.")
    dfs = [_read_dependency_export(p) for p in paths]
    return pd.concat(dfs, ignore_index=True)

def determine_complexity(num_servers: int) -> str:
    if num_servers <= 2:
        return 'Low'
    elif 3 <= num_servers <= 4:
        return 'Medium'
    else:
        return 'Complex'

def group_by_source_all_ports_with_groups(input_grouped_csv_path: str):
    df = pd.read_csv(input_grouped_csv_path, dtype={"Destination port": str})
    df['Source server name'] = df['Source server name'].map(_normalize_text)
    df['Destination server name'] = df['Destination server name'].map(_normalize_text)
    df['Destination port'] = df['Destination port'].astype(str).str.strip()
    edge_ports = defaultdict(set)
    source_to_dests = defaultdict(set)
    for _, r in df.iterrows():
        src = r['Source server name']
        dst = r['Destination server name']
        port = r['Destination port']
        source_to_dests[src].add(dst)
        edge_ports[(src, dst)].add(port)
    rows = []
    for src, dests in source_to_dests.items():
        dest_list = sorted(dests)
        total_servers = 1 + len(dest_list)
        rows.append({'Application Server': src, 'Dependency Servers': dest_list, 'Total Servers': total_servers, 'Complexity': determine_complexity(total_servers)})
    grouped_df = pd.DataFrame(rows, columns=['Application Server', 'Dependency Servers', 'Total Servers', 'Complexity'])
    order = {'Low': 1, 'Medium': 2, 'Complex': 3}
    grouped_df['Complexity Order'] = grouped_df['Complexity'].map(order)
    grouped_df = grouped_df.sort_values(by=['Complexity Order', 'Total Servers', 'Application Server'], ascending=[True, True, True]).drop(columns='Complexity Order').reset_index(drop=True)
    sig_to_group = OrderedDict()
    groups = []
    for _, row in grouped_df.iterrows():
        signature = tuple(row['Dependency Servers'])
        if signature not in sig_to_group:
            sig_to_group[signature] = f"Group {len(sig_to_group) + 1}"
        groups.append(sig_to_group[signature])
    grouped_df.insert(0, 'Group', groups)
    base = os.path.splitext(input_grouped_csv_path)[0]
    complexity_csv_path = base + '_ComplexityGrouping.csv'
    grouped_df.to_csv(complexity_csv_path, index=False)
    app_to_group = {row['Application Server']: row['Group'] for _, row in grouped_df.iterrows()}
    return grouped_df, complexity_csv_path, edge_ports, app_to_group

            edge_key = tuple(sorted([app_node_id, dep_node_id]))
            if edge_key not in edges_dict:
                edges_dict[edge_key] = {"from": app_node_id, "to": dep_node_id, "title": f"Ports: {port_text}", "label": port_text}
    nodes = list(nodes_dict.values())
    edges = list(edges_dict.values())
    options = {
        "interaction": {"hover": True},
        "physics": {"enabled": True, "stabilization": {"enabled": True, "iterations": 200}},
        "configure": {"enabled": True},
        "nodes": {"shape": "dot", "size": 16},

def _load_cmdb(cmdb_path: str) -> pd.DataFrame:
    _, ext = os.path.splitext(cmdb_path.lower())
    if ext == ".csv":
        try:
            df = pd.read_csv(cmdb_path)
        except UnicodeDecodeError:
            df = pd.read_csv(cmdb_path, encoding="ISO-8859-1")
    elif ext in (".xlsx", ".xls"):
        df = pd.read_excel(cmdb_path)
    else:
        raise ValueError("Unsupported CMDB file type. Please use CSV or Excel.")
    if df.empty:
        raise ValueError("CMDB file appears to be empty.")
    cols = list(df.columns)
    col_map = {str(c).strip().lower(): c for c in cols}
    asset_col = None
    for key in col_map:
        if "asset tag" in key or key == "server" or key == "hostname" or "ci name" in key:
            asset_col = col_map[key]
            break
    if asset_col is None and "Asset Tag" in df.columns:
        asset_col = "Asset Tag"
    if asset_col is None:
        raise ValueError("CMDB file must have a column like 'Asset Tag' / 'Server' / 'Hostname'.")
    app_col = None
    for key in col_map:
        if "bsp name" in key or "application" in key or "app name" in key:
            app_col = col_map[key]
            break
    if app_col is None and "BSP Name" in df.columns:
        app_col = "BSP Name"
    if app_col is None:
        raise ValueError("CMDB file must have an application column such as 'BSP Name' / 'Application'.")
    df = df[[asset_col, app_col]].copy()
    df.rename(columns={asset_col: "AssetTag", app_col: "Application"}, inplace=True)
    df["AssetTag_norm"] = df["AssetTag"].astype(str).str.strip().str.upper()
    df["Application"] = df["Application"].astype(str).str.strip()
    df = df.dropna(subset=["AssetTag_norm", "Application"])
    return df

def group_applications_from_edges(edge_ports_map: Dict[Tuple[str, str], set], output_base: str):
    source_to_dests = defaultdict(set)
    for (src_app, dst_app), _ports in edge_ports_map.items():
        source_to_dests[src_app].add(dst_app)
    rows = []
    for src_app, dests in source_to_dests.items():
        dest_list = sorted(dests)
        total = 1 + len(dest_list)
        rows.append({"Application": src_app, "Dependency Applications": dest_list, "Total Applications": total, "Complexity": determine_complexity(total)})
    grouped_df = pd.DataFrame(rows, columns=["Application", "Dependency Applications", "Total Applications", "Complexity"])
    if grouped_df.empty:
        complexity_csv_path = f"{output_base}_AppComplexityGrouping.csv"
        grouped_df.to_csv(complexity_csv_path, index=False)
        return grouped_df, complexity_csv_path, {}
    order = {"Low": 1, "Medium": 2, "Complex": 3}
    grouped_df["Complexity Order"] = grouped_df["Complexity"].map(order)
    grouped_df = grouped_df.sort_values(by=["Complexity Order", "Total Applications", "Application"], ascending=[True, True, True]).drop(columns="Complexity Order").reset_index(drop=True)
    sig_to_group = OrderedDict()
    groups = []
    for _, row in grouped_df.iterrows():
        signature = tuple(row["Dependency Applications"])
        if signature not in sig_to_group:
            sig_to_group[signature] = f"Group {len(sig_to_group) + 1}"
        groups.append(sig_to_group[signature])
    grouped_df.insert(0, "Group", groups)
    complexity_csv_path = f"{output_base}_AppComplexityGrouping.csv"
    grouped_df.to_csv(complexity_csv_path, index=False)
    app_to_group = {row["Application"]: row["Group"] for _, row in grouped_df.iterrows()}
    return grouped_df, complexity_csv_path, app_to_group

def run_full_dep_pipeline_with_cmdb(input_paths: Any, cmdb_path: str, log_func=print) -> Dict[str, str]:
    cwd = os.getcwd()
    paths = [str(input_paths)] if isinstance(input_paths, (str, Path)) else [str(p) for p in input_paths]
    if not paths:
        raise ValueError("No dependency files provided.")
    base_stem = os.path.splitext(os.path.basename(paths[0]))[0] + ("_merged_CMDB" if len(paths) > 1 else "_CMDB")
    output_base = os.path.join(cwd, base_stem)
    log_func(f"[DEP+CMDB] [1/6] Reading {len(paths)} Azure Migrate dependency file(s)...")
    raw_df = _read_dependency_exports(paths)
    log_func(f"[DEP+CMDB]      - Combined rows: {len(raw_df)}")
    log_func(f"[DEP+CMDB] [2/6] Loading CMDB file: {cmdb_path}")
    cmdb_df = _load_cmdb(cmdb_path)
    server_to_app: Dict[str, str] = {}
    for _, row in cmdb_df.iterrows():
        key = _normalize_hostname_for_matching(row["AssetTag"])
        if key:
            server_to_app[key] = row["Application"]
    log_func(f"[DEP+CMDB]      - CMDB entries with application: {len(server_to_app)}")
    raw_df = raw_df.copy()
    raw_df["Source_norm"] = raw_df["Source server name"].map(_normalize_hostname_for_matching)
    raw_df["Dest_norm"] = raw_df["Destination server name"].map(_normalize_hostname_for_matching)
    app_edge_ports = defaultdict(set)
    total_rows = len(raw_df)
    mapped_rows = 0
    for _, r in raw_df.iterrows():
        src_app = server_to_app.get(r["Source_norm"])
        dst_app = server_to_app.get(r["Dest_norm"])
        if src_app and dst_app:
            mapped_rows += 1
            port = str(r["Destination port"]).strip()
            app_edge_ports[(src_app, dst_app)].add(port)
    log_func(f"[DEP+CMDB] [3/6] Mapped {mapped_rows} of {total_rows} rows to application->application edges.")
    if not app_edge_ports:
        raise ValueError("No application-to-application mappings generated. Check CMDB vs server names.")
    edges_rows = [{"Source Application": k[0], "Destination Application": k[1], "Ports": ",".join(sorted(v))} for k, v in app_edge_ports.items()]
    edges_df = pd.DataFrame(edges_rows, columns=["Source Application", "Destination Application", "Ports"])
    edges_csv_path = f"{output_base}_AppEdges.csv"
    edges_df.to_csv(edges_csv_path, index=False)
    log_func(f"[DEP+CMDB]      - App edges CSV: {edges_csv_path}")
    log_func(f"[DEP+CMDB] [4/6] Grouping applications into waves based on connectivity...")
    grouped_df_apps, complexity_csv_path, app_to_group = group_applications_from_edges(app_edge_ports, output_base)
    log_func(f"[DEP+CMDB]      - App complexity CSV: {complexity_csv_path}")
    html_path = ""
    if not grouped_df_apps.empty:
        host_style_df = grouped_df_apps.rename(columns={"Application": "Application Server", "Dependency Applications": "Dependency Servers"})
        log_func(f"[DEP+CMDB] [5/6] Building interactive application network (HTML)...")
        html_path = f"{output_base}_AppNetwork.html"
        create_interactive_network_graph(host_style_df, app_edge_ports, app_to_group, html_path)
        log_func(f"[DEP+CMDB]      - HTML app graph: {html_path}")
    log_func(f"[DEP+CMDB] [6/6] Opening interactive graph in browser...")
    if html_path:
        webbrowser.open('file://' + os.path.abspath(html_path))
    log_func("[DEP+CMDB] === Dependency Mapping DONE (APP-BASED via CMDB) ===")
    return {
        "mode": "WITH_CMDB",
        "input_files": " ; ".join(paths),
        "cmdb_file": cmdb_path,
        "app_edges_csv": edges_csv_path,
        "app_complexity_csv": complexity_csv_path,
        "app_network_html": html_path
    }

def load_rvtools(path: Path) -> Dict[str, pd.DataFrame]:
    xl = pd.ExcelFile(path)
    return {sheet: xl.parse(sheet) for sheet in xl.sheet_names}

def bool_col(df: pd.DataFrame, col_name: str, default: bool = False) -> pd.Series:
    if col_name in df.columns:
        return df[col_name].fillna(False).astype(bool)
    return pd.Series([default] * len(df), index=df.index)

def check_powerstate(vinfo: pd.DataFrame) -> Dict[str, Dict[str, bool]]:
    col = find_col(vinfo, ["powerstate"])
    guest_col = find_col(vinfo, ["guest state"])
    hb_col = find_col(vinfo, ["heartbeat"])
    cfg_col = find_col(vinfo, ["config status"])
    results = {}
    for _, row in vinfo.iterrows():
        vm = str(row["VM"]).strip()
        powered_on = bool(col and str(row[col]).lower() == "poweredon")
        guest_running = bool(guest_col and str(row[guest_col]).lower() == "running")
        hb_ok = bool(hb_col and str(row[hb_col]).lower() in ["green"])
        cfg_ok = bool(cfg_col and str(row[cfg_col]).lower() == "green")
        results[vm] = {"powered_on": powered_on, "guest_running": guest_running, "heartbeat_ok": hb_ok, "config_ok": cfg_ok}
    return results

def check_vmtools(vtools: Optional[pd.DataFrame]) -> Dict[str, Dict[str, Any]]:
    if vtools is None or vtools.empty:
        return {}
    tools_col = find_col(vtools, ["tools"])
    ver_col = find_col(vtools, ["tools version"])
    req_col = find_col(vtools, ["required"])
    results = {}
    for _, row in vtools.iterrows():
        vm = str(row["VM"]).strip()
        tools_state = str(row[tools_col]).lower() if tools_col else ""
        healthy = (tools_state == "toolsok")
        outdated = False
        if ver_col and req_col:
            try:
                outdated = int(row[ver_col]) < int(row[req_col])
            except Exception:
                outdated = False
        results[vm] = {"tools_healthy": healthy and not outdated, "tools_state": tools_state, "tools_outdated": outdated}
    return results

def check_mounted_iso(vcd: Optional[pd.DataFrame], vusb: Optional[pd.DataFrame]) -> Dict[str, bool]:
    results = {}
    if vcd is not None and not vcd.empty:
        for _, row in vcd.iterrows():
            vm = str(row.get("VM", "")).strip()
            connected = bool(row.get("Connected", False)) or bool(row.get("Starts Connected", False))
            if vm and connected:
                results[vm] = True
    if vusb is not None and not vusb.empty:
        for _, row in vusb.iterrows():
            vm = str(row.get("VM", "")).strip()
            if vm and bool(row.get("Connected", False)):
                results[vm] = True
    return results

def check_disk_multiple_1024(vdisk: Optional[pd.DataFrame]) -> Dict[str, Any]:
    if vdisk is None or vdisk.empty:
        return {}
    cap_col = find_col(vdisk, ["capacity mib"])
    if not cap_col:
        return {"_error": "No Capacity column found in vDisk."}
    results: Dict[str, List[Dict[str, Any]]] = {}
    for _, row in vdisk.iterrows():
        vm = str(row.get("VM", "")).strip()
        disk = row.get("Disk")
        try:
            size = int(row[cap_col])
            aligned = (size % 1024 == 0)
        except Exception:
            aligned = False
        if vm and not aligned:
            results.setdefault(vm, []).append({"disk": disk, "size_mib": row[cap_col]})
    return results

def check_partition_free(vpart: Optional[pd.DataFrame], threshold: float = 20.0) -> Dict[str, Any]:
    if vpart is None or vpart.empty:
        return {}
    disk_col = find_col(vpart, ["disk"])
    free_col = find_col(vpart, ["free %"])
    if not disk_col or not free_col:
        return {}
    results: Dict[str, List[Dict[str, Any]]] = {}
    for _, row in vpart.iterrows():
        vm = str(row.get("VM", "")).strip()
        disk = row.get(disk_col)
        try:
            freepct = float(row[free_col])
        except Exception:
            continue
        if vm and freepct < threshold:
            results.setdefault(vm, []).append({"partition": disk, "free_pct": freepct})
    return results

def check_recent_backup(vinfo: pd.DataFrame, max_age_days: int = 7) -> Dict[str, Any]:
    col = find_col(vinfo, ["backup"])
    if not col:
        return {"_error": "Backup info not present in RVTools"}
    results = {}
    now = datetime.now(timezone.utc)
    for _, row in vinfo.iterrows():
        vm = str(row.get("VM", "")).strip()
        val = str(row.get(col, ""))
        if not vm:
            continue
        if val.lower() == "nan" or val.strip() == "":
            results[vm] = {"has_backup": False, "message": "No backup info"}
            continue
        match = re.search(r'at (.*)', val)
        if not match:
            results[vm] = {"has_backup": False, "message": "Unparsable backup info"}
            continue
        try:
            dt = pd.to_datetime(match.group(1), utc=True)
        except Exception:
            results[vm] = {"has_backup": False, "message": "Failed to parse backup date"}
            continue
        age_days = (now - dt).days
        ok = age_days <= max_age_days
        results[vm] = {"has_backup": ok, "age_days": age_days, "backup_datetime": dt}
    return results

def check_snapshots(vsnap: Optional[pd.DataFrame]) -> Dict[str, Any]:
    if vsnap is None or vsnap.empty:
        return {}
    results: Dict[str, List[Dict[str, Any]]] = {}
    for _, row in vsnap.iterrows():
        vm = str(row.get("VM", "")).strip()
        if not vm:
            continue
        results.setdefault(vm, []).append({
            "snapshot_name": row.get("Name"),
            "date": row.get("Date / time"),
            "size_total_mib": row.get("Size MiB (total)")
        })
    return results

def extract_vm_networks_from_rvtools(dfs: Dict[str, pd.DataFrame]) -> Dict[str, List[str]]:
    vnet = dfs.get("vNetwork")
    if vnet is None or vnet.empty:
        return {}
    vnet = vnet.copy()
    vm_col = find_col(vnet, ["vm", "name"])
    if not vm_col:
        return {}
    net_col = None
    for patt in ["portgroup", "port group", "network", "dvportgroup", "dv portgroup"]:
        net_col = find_col(vnet, [patt])
        if net_col:
            break
    if not net_col:
        return {}
    vnet["VM"] = vnet[vm_col].astype(str).str.strip()
    vnet["NetworkName"] = vnet[net_col].astype(str).str.strip()
    vnet = vnet[(vnet["VM"] != "") & (vnet["NetworkName"] != "")]
    vm_to_nets = {}
    for vm, sub in vnet.groupby("VM"):
        nets = sorted({str(x).strip() for x in sub["NetworkName"] if str(x).strip()})
        vm_to_nets[vm] = nets
    return vm_to_nets

def run_rvtools_checks(dfs: Dict[str, pd.DataFrame]):
    vinfo = dfs.get("vInfo")
    if vinfo is None or vinfo.empty:
        raise ValueError("RVTools workbook does not contain a non-empty vInfo sheet.")
    vtools = dfs.get("vTools")
    vcd = dfs.get("vCD")
    vusb = dfs.get("vUSB")
    vdisk = dfs.get("vDisk")
    vpart = dfs.get("vPartition")
    vsnap = dfs.get("vSnapshot")
    vinfo = vinfo.copy()
    vinfo["VM"] = vinfo["VM"].astype(str).str.strip()
    vms = sorted(vinfo["VM"].unique())
    summary = pd.DataFrame(index=vms)
    power = check_powerstate(vinfo)
    tools = check_vmtools(vtools)
    iso = check_mounted_iso(vcd, vusb)
    disks = check_disk_multiple_1024(vdisk)
    free = check_partition_free(vpart)
    backup = check_recent_backup(vinfo)
    snaps = check_snapshots(vsnap)
    summary["Powered On"] = [power.get(vm, {}).get("powered_on", False) for vm in vms]
    summary["Guest Running"] = [power.get(vm, {}).get("guest_running", False) for vm in vms]
    summary["Heartbeat OK"] = [power.get(vm, {}).get("heartbeat_ok", False) for vm in vms]
    summary["Config OK"] = [power.get(vm, {}).get("config_ok", False) for vm in vms]
    summary["VMTools Healthy"] = [tools.get(vm, {}).get("tools_healthy", False) for vm in vms]
    summary["ISO/Device Attached"] = [vm in iso for vm in vms]
    summary["Disk Misaligned"] = [vm in disks for vm in vms if vm != "_error"]
    summary["Low Free Partitions (<20%)"] = [vm in free for vm in vms]
    summary["Backup OK"] = [backup.get(vm, {}).get("has_backup", False) for vm in vms]
    summary["Has Snapshots"] = [vm in snaps for vm in vms]
    return summary, disks, free, iso, snaps

def add_ready_and_reasons(summary: pd.DataFrame) -> pd.DataFrame:
    powered_on = bool_col(summary, "Powered On")
    guest_running = bool_col(summary, "Guest Running")
    heartbeat_ok = bool_col(summary, "Heartbeat OK")
    config_ok = bool_col(summary, "Config OK")
    tools_ok = bool_col(summary, "VMTools Healthy")
    no_iso = ~bool_col(summary, "ISO/Device Attached")
    no_misaligned = ~bool_col(summary, "Disk Misaligned")
    no_lowspace = ~bool_col(summary, "Low Free Partitions (<20%)")
    no_snapshots = ~bool_col(summary, "Has Snapshots")
    summary["ReadyToMigrate"] = (powered_on & guest_running & heartbeat_ok & config_ok & tools_ok & no_iso & no_misaligned & no_lowspace & no_snapshots)
    reasons = []
    for idx in summary.index:
        r = []
        if not powered_on[idx]:
            r.append("VM is not powered on")
        if not guest_running[idx]:
            r.append("Guest OS not running")
        if not heartbeat_ok[idx]:
            r.append("Heartbeat not green")
        if not config_ok[idx]:
            r.append("VM config not healthy")
        if not tools_ok[idx]:
            r.append("VMTools not healthy / outdated")
        if not no_iso[idx]:
            r.append("Mounted ISO or attached device")
        if not no_misaligned[idx]:
            r.append("Disk not multiple of 1024")
        if not no_lowspace[idx]:
            r.append("Low free disk space (<20%)")
        if not no_snapshots[idx]:
            r.append("Snapshots exist")
        reasons.append("; ".join(r) if r else "")
    summary["ReasonNotReady"] = reasons
    return summary

def load_scoped(scope_path: Path, sheet_name: Optional[str] = None) -> pd.DataFrame:
    df = pd.read_excel(scope_path, sheet_name=sheet_name, engine="openpyxl") if sheet_name else pd.read_excel(scope_path, engine="openpyxl")
    if "Server" not in df.columns:
        raise ValueError('Scoped server file must have a column named "Server".')
    df["Server"] = df["Server"].astype(str).str.strip()
    df["Server_UPPER"] = df["Server"].str.upper()
    return df

def run_premigration_pipeline(rvtools_path: str, scope_path: str, out_path: str, scope_sheet: Optional[str] = None, log_func=print) -> Dict[str, Any]:
    rvtools_path_p = Path(rvtools_path)
    scope_path_p = Path(scope_path)
    out_path_p = Path(out_path)
    log_func(f"[PRE] Loading RVTools export: {rvtools_path_p}")
    dfs = load_rvtools(rvtools_path_p)
    vm_networks = extract_vm_networks_from_rvtools(dfs)
    summary_all, disks, free, iso, snaps = run_rvtools_checks(dfs)
    summary_all = add_ready_and_reasons(summary_all)
    log_func(f"[PRE] Loading scoped server list: {scope_path_p}")

    scoped_vm_set = set(scoped_found["VM"].astype(str).str.strip())
    disks_scoped = [(vm, issues) for vm, issues in disks.items() if vm in scoped_vm_set and vm != "_error"]
    free_scoped = [(vm, issues) for vm, issues in free.items() if vm in scoped_vm_set]
    iso_scoped = [(vm, True) for vm in iso if vm in scoped_vm_set]

    scoped_found.drop(columns=["Server_UPPER", "VM_UPPER"], errors="ignore", inplace=True)
    base_pairs = scoped_found[["Server", "VM"]].drop_duplicates()
    net_rows = []
    for _, row in base_pairs.iterrows():

    found_count = scoped_found["Server"].nunique() if "Server" in scoped_found.columns else len(scoped_found)
    log_func("[PRE] === Pre-Migration Health Check DONE ===")

    return {"output_excel": str(out_path_p), "scoped_found": int(len(scoped_found)), "scoped_not_found": int(len(scoped_not_found))}

def _build_vm_resources_from_rvtools(dfs: Dict[str, pd.DataFrame], log_func=print):
    vcpu_df = dfs.get("vCPU")
    vmem_df = dfs.get("vMemory")
    vdisk_df = dfs.get("vDisk")
    if vcpu_df is None or vcpu_df.empty:
        raise ValueError("RVTools missing 'vCPU' sheet.")
    if vmem_df is None or vmem_df.empty:
        raise ValueError("RVTools missing 'vMemory' sheet.")
    if vdisk_df is None or vdisk_df.empty:
        raise ValueError("RVTools missing 'vDisk' sheet.")
    vcpu_df = vcpu_df.copy()
    vmem_df = vmem_df.copy()
    vdisk_df = vdisk_df.copy()
    vcpu_df.columns = [str(c).strip() for c in vcpu_df.columns]
    vmem_df.columns = [str(c).strip() for c in vmem_df.columns]
    vdisk_df.columns = [str(c).strip() for c in vdisk_df.columns]
    vm_col_cpu = find_col(vcpu_df, ["vm", "name"])
    cpu_col = find_col(vcpu_df, ["cpu", "cpus"])
    vm_col_mem = find_col(vmem_df, ["vm", "name"])
    mem_mib_col = find_col(vmem_df, ["size mib", "sizemib"])
    vm_col_disk = find_col(vdisk_df, ["vm", "name"])
    cap_mib_col = find_col(vdisk_df, ["capacity mib", "cap mib"])
    if not all([vm_col_cpu, cpu_col, vm_col_mem, mem_mib_col, vm_col_disk, cap_mib_col]):
        raise ValueError("RVTools columns are missing in vCPU/vMemory/vDisk.")
    vcpu_df["VM"] = vcpu_df[vm_col_cpu].astype(str).str.strip()
    vcpu_df["CPUs"] = pd.to_numeric(vcpu_df[cpu_col], errors="coerce").fillna(0.0)
    cpu_group = vcpu_df.groupby("VM")["CPUs"].sum().reset_index()
    vmem_df["VM"] = vmem_df[vm_col_mem].astype(str).str.strip()
    vmem_df["MemGiB"] = pd.to_numeric(vmem_df[mem_mib_col], errors="coerce").fillna(0.0) / 1024.0
    mem_group = vmem_df.groupby("VM")["MemGiB"].sum().reset_index()
    vdisk_df["VM"] = vdisk_df[vm_col_disk].astype(str).str.strip()
    vdisk_df["ProvTB"] = pd.to_numeric(vdisk_df[cap_mib_col], errors="coerce").fillna(0.0) / (1024.0 * 1024.0)
    used_mib_col = find_col(vdisk_df, ["used mib", "inuse mib", "used mib (total)", "usedmb", "inuse mib (total)"])
    if used_mib_col:
        vdisk_df["UsedTB"] = pd.to_numeric(vdisk_df[used_mib_col], errors="coerce").fillna(0.0) / (1024.0 * 1024.0)
    else:
        vdisk_df["UsedTB"] = 0.0
    disk_group = vdisk_df.groupby("VM")[["ProvTB", "UsedTB"]].sum().reset_index()
    df_vms = cpu_group.merge(mem_group, on="VM", how="outer").merge(disk_group, on="VM", how="outer").fillna(0.0)
    return df_vms, vdisk_df

def _calculate_for_node_type(total_vcpu: float, total_mem_gib: float, total_stor_tb: float, node_type_name: str, profile: Dict[str, float]):
    cores = float(profile["cores"])
    mem_gib = float(profile["mem_gib"])
    raw_tb = float(profile["raw_tb"])
    cpu_overcommit = 4.0
    cpu_headroom = 20.0
    esxi_overhead = 10.0
    mem_headroom = 20.0
    vsan_slack = 25.0
    node_cpu_capacity = cores * cpu_overcommit
    node_cpu_eff = node_cpu_capacity * (1 - cpu_headroom / 100.0)
    node_mem_eff = mem_gib * (1 - esxi_overhead / 100.0) * (1 - mem_headroom / 100.0)
    vsan_eff_r1 = 0.50
    vsan_eff_r5 = 0.75

    def nodes_for_storage(total_tb: float, per_node_raw: float, eff: float, slack: float) -> int:
        usable = per_node_raw * eff * (1 - slack / 100.0)
        if usable <= 0:
            return 0
        return max(1, math.ceil(total_tb / usable))

    nodes_cpu = max(3, math.ceil(total_vcpu / node_cpu_eff)) if node_cpu_eff > 0 else 3
    nodes_mem = max(3, math.ceil(total_mem_gib / node_mem_eff)) if node_mem_eff > 0 else 3
    nodes_stor_r1 = max(3, nodes_for_storage(total_stor_tb, raw_tb, vsan_eff_r1, vsan_slack))
    nodes_stor_r5 = max(4, nodes_for_storage(total_stor_tb, raw_tb, vsan_eff_r5, vsan_slack))

    def limiting(cpu: int, mem: int, stor: int) -> str:
        if cpu >= mem and cpu >= stor:
            return "CPU"
        if mem >= cpu and mem >= stor:
            return "Memory"
        return "Storage"

    rec_r1 = max(nodes_cpu, nodes_mem, nodes_stor_r1)
    rec_r5 = max(nodes_cpu, nodes_mem, nodes_stor_r5)
    result_r1 = dict(NodeType=node_type_name, Scenario="FTT1-RAID1 (Mirroring)", Total_vCPU=round(total_vcpu, 2), Total_Mem_GiB=round(total_mem_gib, 2),
                     Total_Storage_TB=round(total_stor_tb, 2), Nodes_by_CPU=nodes_cpu, Nodes_by_Memory=nodes_mem, Nodes_by_Storage=nodes_stor_r1,
                     Recommended_Nodes=rec_r1, Limiting_Factor=limiting(nodes_cpu, nodes_mem, nodes_stor_r1))
    result_r5 = dict(NodeType=node_type_name, Scenario="FTT1-RAID5 (Erasure Coding)", Total_vCPU=round(total_vcpu, 2), Total_Mem_GiB=round(total_mem_gib, 2),
                     Total_Storage_TB=round(total_stor_tb, 2), Nodes_by_CPU=nodes_cpu, Nodes_by_Memory=nodes_mem, Nodes_by_Storage=nodes_stor_r5,
                     Recommended_Nodes=rec_r5, Limiting_Factor=limiting(nodes_cpu, nodes_mem, nodes_stor_r5))
    return result_r1, result_r5

def _calculate_and_write_assessment(df_vms: pd.DataFrame, vdisk_df: pd.DataFrame, total_vcpu: float, total_mem_gib: float, total_stor_tb: float, out_path: Path, mode_label: str, log_func=print):
    summary_rows = []
    log_func(f"[ASSM] ({mode_label}) In-scope VMs: {len(df_vms)}")
    log_func(f"[ASSM] ({mode_label}) Total vCPU     = {total_vcpu:.2f}")
    log_func(f"[ASSM] ({mode_label}) Total Memory   = {total_mem_gib:.2f} GiB")
    log_func(f"[ASSM] ({mode_label}) Total Storage  = {total_stor_tb:.2f} TB")
    for node_type_name, profile in AVS_NODE_TYPES.items():
        r1, r5 = _calculate_for_node_type(total_vcpu, total_mem_gib, total_stor_tb, node_type_name, profile)
        r1["Scenario"] = f"{r1['Scenario']} - {mode_label}"
        r5["Scenario"] = f"{r5['Scenario']} - {mode_label}"
        summary_rows += [r1, r5]
    df_summary = pd.DataFrame(summary_rows)
    log_func(f"[ASSM] ({mode_label}) Writing AVS capacity assessment to: {out_path}")
    with pd.ExcelWriter(out_path, engine="openpyxl") as writer:
        df_summary.to_excel(writer, index=False, sheet_name="Summary")
        df_vms.to_excel(writer, index=False, sheet_name="PerVM")
        vdisk_df.to_excel(writer, index=False, sheet_name="vDisksRaw")
    return {"output_excel": str(out_path), "summary_rows": int(len(df_summary)), "pervm_rows": int(len(df_vms))}

def run_assessment_pipeline_usedpreferred(rvtools_path: str, out_path: str, log_func=print):

def _recommend_azure_size(req_vcpu: float, req_mem_gib: float, allowed_families: set) -> Optional[Dict[str, Any]]:
    req_vcpu_int = int(math.ceil(req_vcpu))
    candidates = [s for s in AZURE_SIZE_CATALOG if s["family"] in allowed_families and s["vcpus"] >= req_vcpu_int and s["mem_gib"] >= float(req_mem_gib)]
    if not candidates:
        return None
    candidates.sort(key=lambda x: (x["vcpus"], x["mem_gib"]))
    return candidates[0]

def run_azure_vm_assessment_pipeline(rvtools_path: str, out_path: str, family_mode: str = "AUTO", headroom_pct: float = 20.0, log_func=print) -> Dict[str, Any]:
    if headroom_pct < 0:
        raise ValueError("Headroom % must be >= 0")
    allowed = {"Dsv5", "Esv5"} if family_mode.upper() == "AUTO" else ({"Dsv5"} if family_mode.upper() == "DSV5" else {"Esv5"})
    dfs = load_rvtools(Path(rvtools_path))
    df = _build_vm_cpu_mem_from_rvtools(dfs)
    if df.empty:
        raise ValueError("No VMs found in vCPU/vMemory sheets.")
    df["HeadroomPct"] = headroom_pct
    df["Req_vCPU"] = df["vCPU"] * (1.0 + headroom_pct / 100.0)
    df["Req_MemGiB"] = df["MemGiB"] * (1.0 + headroom_pct / 100.0)
    rec_names, rec_fams, rec_vcpus, rec_mem, notes = [], [], [], [], []
    for _, r in df.iterrows():
        rec = _recommend_azure_size(r["Req_vCPU"], r["Req_MemGiB"], allowed)
        if rec is None:
            rec_names.append("")
            rec_fams.append("")
            rec_vcpus.append(None)
            rec_mem.append(None)
            notes.append("NO_MATCH_IN_CATALOG (extend AZURE_SIZE_CATALOG)")
        else:
            rec_names.append(rec["name"])
            rec_fams.append(rec["family"])
            rec_vcpus.append(rec["vcpus"])
            rec_mem.append(rec["mem_gib"])
            notes.append("")
    df["RecommendedSize"] = rec_names
    df["RecommendedFamily"] = rec_fams
    df["Recommended_vCPU"] = rec_vcpus
    df["Recommended_MemGiB"] = rec_mem
    df["Notes"] = notes
    summary = (df[df["RecommendedSize"].astype(str).str.strip() != ""]
               .groupby(["RecommendedFamily", "RecommendedSize"]).size()
               .reset_index(name="VMCount")
               .sort_values(["RecommendedFamily", "VMCount"], ascending=[True, False]))
    not_mapped = df[df["RecommendedSize"].astype(str).str.strip() == ""].copy()
    catalog_df = pd.DataFrame(AZURE_SIZE_CATALOG)
    out_path_p = Path(out_path)
    log_func(f"[AZURE] Writing Azure VM assessment to: {out_path_p}")
    with pd.ExcelWriter(out_path_p, engine="openpyxl") as writer:
        summary.to_excel(writer, sheet_name="Summary", index=False)
        df.to_excel(writer, sheet_name="PerVM", index=False)
        not_mapped.to_excel(writer, sheet_name="NotMapped", index=False)
        catalog_df.to_excel(writer, sheet_name="CatalogUsed", index=False)
    return {"output_excel": str(out_path_p), "vms_total": int(len(df)), "vms_sized": int(len(df) - len(not_mapped)), "vms_not_mapped": int(len(not_mapped))}

def run_ping(server: str, count: int = 2, timeout: int = 3):
    if os.name == "nt":
        cmd = ["ping", "-n", str(count), "-w", str(timeout * 1000), server]
    else:
        cmd = ["ping", "-c", str(count), "-W", str(timeout), server]
    try:
        completed = subprocess.run(cmd, capture_output=True, text=True, timeout=timeout * count + 5)
        success = (completed.returncode == 0)
        output = (completed.stdout or "") + "\n" + (completed.stderr or "")
    except Exception as e:
        success = False
        output = f"ERROR running ping: {e}"
    return success, output

def run_traceroute(server: str, max_hops: int = 20, timeout: int = 60):
    if os.name == "nt":
        cmd = ["tracert", "-d", "-h", str(max_hops), server]
    else:
        cmd = ["traceroute", "-n", "-m", str(max_hops), server]
    try:
        completed = subprocess.run(cmd, capture_output=True, text=True, timeout=timeout)
        success = (completed.returncode == 0)
        output = (completed.stdout or "") + "\n" + (completed.stderr or "")
    except Exception as e:
        success = False
        output = f"ERROR running traceroute: {e}"
    return success, output

def run_postmigration_pipeline(scope_path: str, out_path: str, log_func=print) -> Dict[str, Any]:
    scoped = load_scoped(Path(scope_path), sheet_name=None)
    servers = sorted(set(scoped["Server"].astype(str).str.strip()))
    servers = [s for s in servers if s]
    rows = []
    for server in servers:
        log_func(f"[POST] Pinging {server} ...")
        ping_ok, ping_out = run_ping(server)
        log_func(f"[POST] Traceroute {server} ...")
        trace_ok, trace_out = run_traceroute(server)
        rows.append({"Server": server, "PingSuccess": ping_ok, "TracerouteSuccess": trace_ok, "PingOutput": ping_out, "TracerouteOutput": trace_out})
    df = pd.DataFrame(rows)
    out_path_p = Path(out_path)
    out_path_p.parent.mkdir(parents=True, exist_ok=True)
    with pd.ExcelWriter(out_path_p, engine="openpyxl") as writer:
        df.to_excel(writer, sheet_name="PostMigrationStatus", index=False)
    success_count = sum(1 for r in rows if r["PingSuccess"])
    return {"output_excel": str(out_path_p), "servers_total": int(len(servers)), "ping_success": int(success_count)}

    # ------------------------------
    # Queue / logging
    # ------------------------------
    def _enqueue_log(self, msg: str):
        self.msg_queue.put(("log", msg))

    def _enqueue_call(self, func, *args, **kwargs):
        self.msg_queue.put(("call", (func, args, kwargs)))

    def _drain_queue(self):
        try:
            while True:
                kind, payload = self.msg_queue.get_nowait()
                if kind == "log":
                    self._log_to_text(payload)
                elif kind == "call":
                    func, args, kwargs = payload
                    func(*args, **kwargs)
        except queue.Empty:
            pass
        self.after(100, self._drain_queue)

    def _log_to_text(self, msg: str):
        self.log.configure(state="normal")
        self.log.insert(tk.END, msg + "\n")
        self.log.see(tk.END)
        self.log.configure(state="disabled")

    def _on_drop_dep(self, event):
        data = event.data
        paths = []
        if data.strip().startswith("{"):
            parts, buf, brace = [], "", False
            for ch in data:
                if ch == "{": brace, buf = True, ""
                elif ch == "}":
                    brace = False
                    if buf: parts.append(buf); buf = ""
                elif brace: buf += ch
                elif ch == " " and not brace:
                    if buf: parts.append(buf); buf = ""
                else: buf += ch
            if buf: parts.append(buf)
            for p in parts:
                if os.path.isfile(p): paths.append(p)
        else:
            for token in data.split():
                if os.path.isfile(token): paths.append(token)
        if paths:
            self._set_dep_selected(paths)
        else:
            messagebox.showerror("Invalid file(s)", f"Couldn't read dropped item(s):\n{data}")

    def _browse_dep(self):
        fps = filedialog.askopenfilenames(title="Select Azure Migrate Dependency Export(s)", filetypes=[("CSV files", "*.csv"), ("Excel files", "*.xlsx *.xls"), ("All files", "*.*")])
        if fps: self._set_dep_selected(list(fps))

    def _browse_cmdb(self):
        fp = filedialog.askopenfilename(title="Select CMDB file (Asset Tag + Application)", filetypes=[("CSV files", "*.csv"), ("Excel files", "*.xlsx *.xls"), ("All files", "*.*")])
        if fp: self.cmdb_path_var.set(fp)

    def _run_dep_pipeline_threaded(self):
        paths = list(self.dep_files)
        if not paths:
            raw = self.file_path.get().strip()
            if raw:
                candidates = [p.strip() for p in raw.split(";") if p.strip()]
                paths = [p for p in candidates if os.path.isfile(p)]
        if not paths:
            messagebox.showerror("No file", "Please select one or more Azure Migrate dependency files first."); return
        mode = self.dep_mode_var.get()
        cmdb_path = self.cmdb_path_var.get().strip()
        if mode == "WITH_CMDB":
            if not cmdb_path or not os.path.isfile(cmdb_path):
                messagebox.showerror("CMDB file", "Please select a valid CMDB file for Application-based mode."); return
        self.run_dep_btn.configure(state="disabled")
        self.dep_progress.start(15)
        self._enqueue_log(f"[DEP] Starting dependency mapping in mode: {mode} ({len(paths)} file(s))")
        def work():
            try:
                outputs = run_full_dep_pipeline_with_cmdb(paths, cmdb_path, log_func=self._enqueue_log) if mode == "WITH_CMDB" else run_full_dep_pipeline(paths, log_func=self._enqueue_log)
                self._enqueue_call(self._on_dep_success, outputs)
            except Exception as e:
                self._enqueue_call(self._on_dep_error, str(e))
        threading.Thread(target=work, daemon=True).start()

    def _on_dep_success(self, outputs: dict):
        self.dep_progress.stop()
        self.run_dep_btn.configure(state="normal")
        self._enqueue_log("[DEP] Outputs:")
        for k, v in outputs.items(): self._enqueue_log(f"[DEP]  - {k}: {v}")
        messagebox.showinfo("Dependency Mapping", "Pipeline completed.\nInteractive graph should be open in your browser.")

    def _on_dep_error(self, msg: str):
        self.dep_progress.stop()
        self.run_dep_btn.configure(state="normal")
        self._enqueue_log(f"[DEP] ERROR: {msg}")
        messagebox.showerror("Dependency Mapping Error", msg)

        if ids: self._enqueue_log(f"[NE] Filter NE IDs: {ids}")
        if map_path: self._enqueue_log(f"[NE] VLAN mapping file: {map_path}")
        def work():
            try:
                outputs = run_ne_status_script(server, user, pwd, outdir, ne_ids=ids, name_patterns=name_pats, log_func=self._enqueue_log)

            filetypes=[("CSV/Excel", "*.csv *.xlsx *.xls"), ("All files", "*.*")]
        )
        if fps:
            self.nsx_files = list(fps)
            self.nsx_files_var.set(" ; ".join(self.nsx_files))

    def _clear_nsx_files(self):
        self.nsx_files = []
        self.nsx_files_var.set("")

        d = filedialog.askdirectory(title="Select output folder")
        if d: self.nsx_out_var.set(d)

    def _run_nsx_pipeline_threaded(self):
        files = self.nsx_files or [p.strip() for p in self.nsx_files_var.get().split(";") if p.strip()]
        files = [p for p in files if os.path.isfile(p)]
        if not files:
            messagebox.showerror("Inputs", "Select at least one NSX-T export file."); return
        out_base = self.nsx_out_var.get().strip()
        if not out_base:
            messagebox.showerror("Output", "Select an output folder."); return
        try:
            thr = int(float(self.nsx_threshold_var.get().strip()))
        except Exception:
            messagebox.showerror("Threshold", "Threshold must be a number (e.g., 20)."); return

        run_dir = str(Path(out_base) / f"NSXT_to_NSG_CSV_{datetime.now().strftime('%Y%m%d_%H%M%S')}")
        pass_only = bool(self.nsx_pass_only_var.get())
        self.run_nsx_btn.configure(state="disabled")
        self.nsx_progress.start(12)
        self._enqueue_log(f"[NSX] Output: {run_dir}")
        for f in files: self._enqueue_log(f"[NSX] Input : {f}")

        def work():
            try:
                outputs = run_nsx_csv_analysis(files, run_dir, pass_only, thr, log=self._enqueue_log)
                self._enqueue_call(self._on_nsx_success, outputs, run_dir)
            except Exception as e:
                self._enqueue_call(self._on_nsx_error, str(e))
        threading.Thread(target=work, daemon=True).start()

    def _on_nsx_success(self, outputs: Dict[str, str], run_dir: str):
        self.nsx_progress.stop()
        self.run_nsx_btn.configure(state="normal")
        self._enqueue_log("[NSX] Done. Files:")
        for k, v in outputs.items(): self._enqueue_log(f"  - {k}: {v}")
        messagebox.showinfo("NSX-T to NSG", f"Completed.\n\n{run_dir}")

    def _on_nsx_error(self, msg: str):
        self.nsx_progress.stop()
        self.run_nsx_btn.configure(state="normal")
        self._enqueue_log(f"[NSX] ERROR: {msg}")
        messagebox.showerror("NSX-T to NSG Error", msg)
